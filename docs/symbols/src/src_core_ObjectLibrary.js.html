  3 
  4 var UNIVERSE = UNIVERSE || {};
  5 
  6 UNIVERSE.ObjectLibrary = function () {
  7     var objects = [],
  8         numberOfElements = 0;
  9 
 10     // adds a mesh object to the object library
 11     // id -> unique id of the object
 12     // url -> url used to retrieve the json of the model
 13     // material -> material to apply to the model's geometry
 14     this.addGeometryObjectFromUrl = function (id, url, callback) {
 15         // if we have already loaded an onject with this id, return
 16         if (objects[id]) {
 17             callback();
 18             return;
 19         }
 20 
 21         // Have to do this to avoid a race condition and avoid requesting it every time
 22         objects[id] = "loading";
 23 
 24         // use a JSON loader to load the mesh from JSON
 25         var jsonLoader = new THREE.JSONLoader();
 26         jsonLoader.load(
 27             url,
 28             function (geometry) {
 29                 //var mesh = new THREE.Mesh(geometry, material);
 30 
 31                 // add the object to our list of objects
 32                 objects[id] = geometry;
 33                 //numberOfElements++;
 34                 //console.log("objects after add: " + JSON.stringify(objects));
 35                 //console.log("numberOfElements after add: " + JSON.stringify(numberOfElements))
 36 
 37                 // execute the callback
 38                 callback();
 39             }
 40         );
 41 
 42     };
 43 
 44     // gets an object from the library based on the given id
 45     this.getObjectById = function (id, callback) {
 46         //console.log("number of elements: " + numberOfElements);
 47         var object = objects[id],
 48             objectLib;
 49 
 50         if (!object) {
 51             throw "Tried to retrieve object [" + id + "] from object library but didn't exist";
 52         } else if (object === "loading") {
 53             objectLib = this;
 54             setTimeout(function () {
 55                 objectLib.getObjectById(id, callback);
 56             }, 1000);
 57         } else {
 58             callback(object);
 59         }
 60     };
 61 
 62     this.setObject = function (id, object) {
 63         objects[id] = object;
 64     };
 65 };
