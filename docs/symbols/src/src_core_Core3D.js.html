129     function render() {
130         zoom(curZoomSpeed);
131 
132         //console.log("target: " + JSON.stringify(target));
133         rotation.x += (target.x - rotation.x) * 0.1;
134         rotation.y += (target.y - rotation.y) * 0.1;
135         distance += (self.distanceTarget - distance) * 0.3;
136 
137         camera.position.x = distance * Math.sin(rotation.x) * Math.cos(rotation.y);
138         camera.position.y = distance * Math.sin(rotation.y);
139         camera.position.z = distance * Math.cos(rotation.x) * Math.cos(rotation.y);
140         camera.lookAt(scene.position);
141 
142         vector.copy(camera.position);
143 
144         scaleDrawnObjects();
145 
146         renderer.clear();
147         renderer.render(scene, camera);
148     }

150     function scaleDrawnObjects() {
151 	    var i,
152 	        objectPosition,
153 	        distanceFromCamera,
154 	        scaleFactor;
155         for (i in drawnObjects) {
156             if (drawnObjects[i].scale === true) {
157                 objectPosition = drawnObjects[i].shape.position;
158                 distanceFromCamera = objectPosition.distanceTo(camera.position);
159                 scaleFactor = distanceFromCamera / (6371 * 7);
160                 drawnObjects[i].shape.scale.x = drawnObjects[i].shape.scale.y = drawnObjects[i].shape.scale.z = scaleFactor;
161             }
162         }
163     }

166     function onMouseDown(event) {
167         event.preventDefault();
168 
169         container.addEventListener('mousemove', onMouseMove, false);
170         container.addEventListener('mouseup', onMouseUp, false);
171         container.addEventListener('mouseout', onMouseOut, false);
172 
173         mouseOnDown.x = -event.clientX;
174         mouseOnDown.y = event.clientY;
175 
176         targetOnDown.x = target.x;
177         targetOnDown.y = target.y;
178 
179         container.style.cursor = 'move';
180     }

182     function onMouseMove(event) {
183         mouse.x = -event.clientX;
184         mouse.y = event.clientY;
185 
186         var zoomDamp = distance / (35000);
187 
188         target.x = targetOnDown.x + (mouse.x - mouseOnDown.x) * 0.005 * zoomDamp;
189         target.y = targetOnDown.y + (mouse.y - mouseOnDown.y) * 0.005 * zoomDamp;
190 
191         target.y = target.y > PI_HALF ? PI_HALF : target.y;
192         target.y = target.y < -PI_HALF ? -PI_HALF : target.y;
193     }

234     // This: http://stackoverflow.com/questions/55677/how-do-i-get-the-coordinates-of-a-mouse-click-on-a-canvas-element
235 	function relMouseCoords(event) {
236 		if (event.offsetX !== undefined && event.offsetY !== undefined) { 
237 			return {x: event.offsetX, y: event.offsetY}; 
238 		}
239 	    var totalOffsetX = 0;
240 	    var totalOffsetY = 0;
241 	    var canvasX = 0;
242 	    var canvasY = 0;
243 	    var currentElement = this;
244 
245 	    do{
246 	        totalOffsetX += currentElement.offsetLeft - currentElement.scrollLeft;
247 	        totalOffsetY += currentElement.offsetTop - currentElement.scrollTop;
248 	    }
249 	    while(currentElement = currentElement.offsetParent)
250 
251 	    canvasX = event.pageX - totalOffsetX;
252 	    canvasY = event.pageY - totalOffsetY;
253 
254 	    return {x:canvasX, y:canvasY}
255 	}

258 		var i,
259 		    j,
260 		    intersectedObjects = [];
261 		
262 		for (j = 0; j < intersects.length; j += 1) {
263 			intersectedObjects.push(getObjectForShape(intersects[j]));
264 		}
265 
266 		for (i = 0; i < intersectionListeners.length; i += 1) {
267 			intersectionListeners[i](intersectedObjects);
268 		}
269 	}

311     function onWindowResize() {
312         // so right now this event was fired when the entire window was resized,
313         // but the individual dom elements haven't been resized yet. We will wait
314         // a bit then execute the actual resize code so we can use the updated
315         // element sizes.
316         clearTimeout(resizeTimeout);
317         resizeTimeout = setTimeout(function () {
318             resize();
319         }, 250);
320     }

330     function zoom(delta) {
331         self.distanceTarget -= delta;
332         self.distanceTarget = self.distanceTarget > self.maxZoom ? self.maxZoom : self.distanceTarget;
333         self.distanceTarget = self.distanceTarget < self.minZoom ? self.minZoom : self.distanceTarget;
334     }

397     this.moveCameraTo = function (position_vector) {
398         // This method converts a position into the rotation coordinate system used to move the camera
399         // The target.x parameter is the rotation angle from the positive Z axis
400         // target.y is the rotation angle away from the z-x plane
401 
402 
403         var cameraVector = new THREE.Vector3(),
404             zAxisVector = new THREE.Vector3(0, 0, 1), // unit vector along the z axis
405             yAxisVector = new THREE.Vector3(0, 1, 0), // unit vector along the y axis
406             positionY0Vector = new THREE.Vector3(),
407             y, // The angle between the positive y axis and the input position vector
408             x; // the angle between the input vector projected on the z-x plane and the z-axis
409 
410         // copy so we don't stomp on the original
411         cameraVector.copy(position_vector);
412 
413         // sets the distance from the center of the scene the camera will end up
414         distanceTarget = cameraVector.length();
415 
416         // vector that removes the y component of the target vector for purpose of calculating the angle
417         // between it the input position_vector and the y-z plane
418         positionY0Vector.copy(cameraVector);
419 
420         // set the y to zero and normalize to unit length
421         positionY0Vector.y = 0;
422         positionY0Vector.normalize();
423 
424         //normalize the position_vector to unit length
425         cameraVector.normalize();
426 
427         // calculates the angle between the positive y axis and the input position vector
428         // then subtract this from 90 degrees to shift it to be from the z-x plane
429         y = (Math.PI / 2) - Math.acos(yAxisVector.dot(cameraVector));
430 
431         // calculate the angle between the input vector projected on the z-x plane and the z-axis
432         x = Math.acos(zAxisVector.dot(positionY0Vector));
433 
434         // since the above calculation will return between 0 and 180 degrees, invert it if we're in the
435         // negative x direction
436         if (positionY0Vector.x < 0) {
437             x = -x;
438         }
439 
440         // set it to zero if NaN
441         target.y = isNaN(y) ? 0 : y;
442         target.x = isNaN(x) ? 0 : x;
443     };

494     function updateGamepadStatus() {
495 
496         // right stick
497 
498         // adjust the rotation scaling factor by how close we are to the earth.
499         // when we are far away from the earth, we want to rotate faster than
500         // when we are closer to the earth.
501         var centerPoint = new THREE.Vector3(0, 0, 0),
502             distanceFromEarth = centerPoint.distanceTo(camera.position),
503             adjustedRotationScalingFactor = GAMEPAD_CAMERA_ROTATION_MIN_SCALING_FACTOR +
504                 ((distanceFromEarth / self.maxZoom) * GAMEPAD_CAMERA_ROTATION_DELTA),
505             xRot = gamepad.axes.Right_Stick_X,
506             yRot = gamepad.axes.Right_Stick_Y,
507             zoomInAmount,
508             zoomOutAmount;
509 
510         xRot = xRot > -GAMEPAD_CAMERA_MIN_SENSITIVITY && xRot < GAMEPAD_CAMERA_MIN_SENSITIVITY ?
511                 0 : xRot * adjustedRotationScalingFactor * -1;
512 
513         yRot = yRot > -GAMEPAD_CAMERA_MIN_SENSITIVITY && yRot < GAMEPAD_CAMERA_MIN_SENSITIVITY ?
514                 0 : yRot * adjustedRotationScalingFactor;
515 
516         self.addRotationToCameraTarget(xRot, yRot);
517 
518         // TODO: left stick
519         /*
520         var xPan = gamepad.axes["Left_Stick_X"];
521         xPan = xPan > -GAMEPAD_CAMERA_MIN_SENSITIVITY && xPan < GAMEPAD_CAMERA_MIN_SENSITIVITY
522             ? 0 : xPan * GAMEPAD_PAN_SCALING_FACTOR;
523 
524         var yPan = gamepad.axes["Left_Stick_Y"];
525         yPan = yPan > -GAMEPAD_CAMERA_MIN_SENSITIVITY && yPan < GAMEPAD_CAMERA_MIN_SENSITIVITY
526             ? 0 : yPan * GAMEPAD_PAN_SCALING_FACTOR;
527 
528         camera.position.x += xPan;
529         camera.position.y += yPan;
530         //console.log("cam: " + JSON.stringify(camera.position) + " mv " + xPan + ", " + yPan);
531         */
532 
533         // zoom in / out
534         zoomInAmount = gamepad.buttons.Right_Trigger_2;
535         if (zoomInAmount > GAMEPAD_ZOOM_MIN_SENSITIVITY) {
536             zoom(zoomInAmount * GAMEPAD_ZOOM_SCALING_FACTOR);
537         }
538 
539         zoomOutAmount = gamepad.buttons.Left_Trigger_2;
540         if (zoomOutAmount > GAMEPAD_ZOOM_MIN_SENSITIVITY) {
541             zoom(zoomOutAmount * -GAMEPAD_ZOOM_SCALING_FACTOR);
542         }
543 
544         // setup the callback
545         window.requestAnimFrame(updateGamepadStatus);
546     }
