 28 // // call with a set of points on the earth and a sensor origin
 29 // points is an array with .x, .y. z and the vehicle origin is also an object with just an .x, .y and .z
 30 // Assume that passed in points are all in THREE.js
 31 UNIVERSE.SensorProjectionGeometry = function (sensorOrigin, groundPoints) {
 32 
 33     THREE.Geometry.call(this);
 34 
 35     var segmentsY = 1,
 36         x,
 37         y,
 38         vertices = [],
 39         uvs = [],
 40         segmentsX,
 41         xpos,
 42         ypos,
 43         zpos,
 44         u,
 45         v = 0,
 46         verticesRow = [],
 47         uvsRow = [],
 48         v1,
 49         v2,
 50         v3,
 51         v4,
 52         n1,
 53         n2,
 54         n3,
 55         n4,
 56         uv1,
 57         uv2,
 58         uv3,
 59         uv4;
 60 
 61     // tack on the initial ground point back to the end of the groundpoints array
 62     groundPoints.push(groundPoints[0]);
 63     segmentsX = groundPoints.length - 1;
 64 
 65     // Create the top points around the vehicle
 66 
 67     for (x = 0; x <= segmentsX; x += 1) {
 68         u = x / segmentsX;
 69 
 70         xpos = sensorOrigin.x;
 71         ypos = sensorOrigin.y;
 72         zpos = sensorOrigin.z;
 73 
 74         //console.log("Vertice point: " + xpos + "," + ypos + "," + zpos);
 75         this.vertices.push(new THREE.Vertex(new THREE.Vector3(xpos, ypos, zpos)));
 76         verticesRow.push(this.vertices.length - 1);
 77         uvsRow.push(new THREE.UV(u, v));
 78     }
 79 
 80     vertices.push(verticesRow);
 81     uvs.push(uvsRow);
 82 
 83     // Now create the ground lines
 84     v = 1;
 85     verticesRow = [];
 86     uvsRow = [];
 87     for (x = 0; x <= segmentsX; x += 1) {
 88         u = x / segmentsX;
 89 
 90         xpos = groundPoints[x].x;
 91         ypos = groundPoints[x].y;
 92         zpos = groundPoints[x].z;
 93 
 94         //        console.log("Vertice point: " + xpos + "," + ypos + "," + zpos);
 95         this.vertices.push(new THREE.Vertex(new THREE.Vector3(xpos, ypos, zpos)));
 96         verticesRow.push(this.vertices.length - 1);
 97         uvsRow.push(new THREE.UV(u, v));
 98 
 99     }
100     vertices.push(verticesRow);
101     uvs.push(uvsRow);
102 
103     for (y = 0; y < segmentsY; y += 1) {
104 
105         for (x = 0; x < segmentsX; x += 1) {
106 
107             v1 = vertices[y][x];
108             v2 = vertices[y + 1][x];
109             v3 = vertices[y + 1][x + 1];
110             v4 = vertices[y][x + 1];
111 
112             // FIXME: These normals aren't right for cones.
113 
114             n1 = this.vertices[v1].position.clone().setY(0).normalize();
115             n2 = this.vertices[v2].position.clone().setY(0).normalize();
116             n3 = this.vertices[v3].position.clone().setY(0).normalize();
117             n4 = this.vertices[v4].position.clone().setY(0).normalize();
118 
119             uv1 = uvs[y][x].clone();
120             uv2 = uvs[y + 1][x].clone();
121             uv3 = uvs[y + 1][x + 1].clone();
122             uv4 = uvs[y][x + 1].clone();
123 
124             this.faces.push(new THREE.Face4(v1, v2, v3, v4, [n1, n2, n3, n4]));
125             this.faceVertexUvs[0].push([uv1, uv2, uv3, uv4]);
126         }
127 
128     }
129 
130     this.computeCentroids();
131     this.computeFaceNormals();
132     this.dynamic = true;
133 };
134 UNIVERSE.SensorProjectionGeometry.prototype = new THREE.Geometry();
135 UNIVERSE.SensorProjectionGeometry.prototype.constructor = UNIVERSE.SensorProjectionGeometry;

138 UNIVERSE.SensorProjectionGeometry.prototype.recalculateVertices = function (sensorOrigin, groundPoints) {
139 
140     groundPoints.push(groundPoints[0]);
141 
142     this.dynamic = true;
143     // Create the top points around the vehicle
144     var segmentsX = groundPoints.length,
145         segmentsY = 1,
146         xpos,
147         ypos,
148         zpos,
149         u,
150         x;
151     // tack on the initial ground point back to the end of the groundpoints array
152 
153     for (x = 0; x < segmentsX; x += 1) {
154         u = x / segmentsX;
155 
156         xpos = sensorOrigin.x;
157         ypos = sensorOrigin.y;
158         zpos = sensorOrigin.z;
159 
160         this.vertices[x].position = {
161             x: xpos,
162             y: ypos,
163             z: zpos
164         };
165     }
166 
167     // Now create the ground lines
168     for (x = 0; x < segmentsX; x += 1) {
169         u = x / segmentsX;
170 
171         xpos = groundPoints[x].x;
172         ypos = groundPoints[x].y;
173         zpos = groundPoints[x].z;
174 
175         this.vertices[x + segmentsX].position = {
176             x: xpos,
177             y: ypos,
178             z: zpos
179         };
180     }
181 
182 
183     this.__dirtyVertices = true;
184 };
