  9 var UNIVERSE = UNIVERSE || {};
 10 
 11 UNIVERSE.PropogationLine = function (object, universe, earthExtensions, material, geometry) {
 12     var timeToPropogate = new Date(universe.getCurrentUniverseTime()),
 13         loopCount = 1440,
 14         eciLocations = [],
 15         j,
 16         location,
 17         convertedLocation,
 18         vector,
 19         THREE,
 20         lineS,
 21         lineGraphicsObject;
 22 
 23     // draw a vertex for each minute in a 24 hour period
 24     // dropped this to a vertex for every 5 minutes.  This seems to be about the max that you can use for a LEO
 25     // and still look decent.  HEOs and GEOs look fine with much greater spans.  For performance reasons, may want
 26     // to make this a param that can be set per vehicle
 27     for (j = 0; j < loopCount; j += 5) {
 28         location = object.propagator(timeToPropogate, false);
 29         eciLocations.push(location);
 30         convertedLocation = Utilities.eciTo3DCoordinates(location, earthExtensions);
 31         if (convertedLocation) {
 32             vector = new THREE.Vector3(convertedLocation.x, convertedLocation.y, convertedLocation.z);
 33             geometry.vertices.push(new THREE.Vertex(vector));
 34         }
 35 
 36         timeToPropogate.setMinutes(timeToPropogate.getMinutes() + 5);
 37     }
 38 
 39     lineS = new THREE.Line(geometry, material);
 40 
 41     lineGraphicsObject = new UNIVERSE.GraphicsObject(
 42         object.id + "_propogation",
 43         object.objectName,
 44         undefined,
 45         function (elapsedTime) {
 46             if (earthExtensions.lockCameraToWithEarthRotation) {
 47                 var length = eciLocations.length,
 48                     i,
 49                     convertedLocation;
 50                 for (i = 0; i < length; i += 1) {
 51                     convertedLocation = Utilities.eciTo3DCoordinates(eciLocations[i], earthExtensions);
 52                     if (convertedLocation && lineS.geometry.vertices[i]) {
 53                         lineS.geometry.vertices[i].position = {
 54                             x: convertedLocation.x,
 55                             y: convertedLocation.y,
 56                             z: convertedLocation.z
 57                         };
 58                     }
 59                 }
 60                 lineS.geometry.__dirtyVertices = true;
 61             }
 62 
 63         },
 64         function () {
 65             universe.draw(this.id, lineS, false);
 66         }
 67     );
 68 
 69     return lineGraphicsObject;
 70 };
