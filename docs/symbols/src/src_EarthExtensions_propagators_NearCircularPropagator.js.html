  2 /*global MathTools, Constants */
  3 var NearCircularPropagator = {
  4 
  5     propagateOrbit: function (kep, elapsedTime, dt, timeAtStartOfPropagation) {
  6         //console.log('keping it up');
  7         var MA = kep.getMeanAnomaly() + kep.getMeanMotion() * elapsedTime, //update the mean anomaly (deg)
  8             w = MathTools.toRadians(kep.getArgOfPerigee()),
  9             ra = MathTools.toRadians(kep.getRaan()),
 10             inc = MathTools.toRadians(kep.getInclination()),
 11             ecc = kep.getEccentricity(),
 12             mu = Constants.muEarth,
 13             EA,
 14             errorThreshold = 1e-5,  //how accurately do we need to solve for the eccentric anomaly?
 15             i,
 16             myerror,
 17             f,
 18             p,
 19             r,
 20             h,
 21             x,
 22             y,
 23             z,
 24             xdot,
 25             ydot,
 26             zdot,
 27             eciState = new UNIVERSE.ECICoordinates();
 28 
 29         MA = MathTools.toRadians(MA);  //convert the mean anomaly to radians
 30         //iterate to solve for the eccentric anomaly
 31         EA = MA * 0.95; //fist guess at the eccentric anomaly (rads)
 32 
 33         for (i = 0; i < 500; i += 1) {
 34             myerror = MA - (EA - ecc * Math.sin(EA));
 35             if (Math.abs(myerror) > errorThreshold) {
 36                 if (myerror > 0) {
 37                     EA = EA + Math.abs(MA - EA) / 2;
 38                 } else {
 39                     if (myerror < 0) {
 40                         EA = EA - Math.abs(MA - EA) / 2;
 41                     }
 42                 }
 43             } else {
 44                 break;
 45             }
 46 
 47         }
 48 
 49         f = 2 * Math.atan(Math.sqrt((1 + ecc) / (1 - ecc)) * Math.tan(EA / 2));
 50         p = kep.getSemimajorAxis() * (1 - ecc * ecc);
 51         r = kep.getSemimajorAxis() * (1 - ecc * Math.cos(EA)); //radius
 52         h = Math.sqrt(mu * kep.getSemimajorAxis() * (1 - ecc * ecc));
 53         x = r * (Math.cos(ra) * Math.cos(w + f) - Math.sin(ra) * Math.sin(w + f) * Math.cos(inc));
 54         y = r * (Math.sin(ra) * Math.cos(w + f) + Math.cos(ra) * Math.sin(w + f) * Math.cos(inc));
 55         z = r * Math.sin(inc) * Math.sin(w + f);
 56         xdot = ((x * h * ecc) / (r * p)) * Math.sin(f) - (h / r) * (Math.cos(ra) * Math.sin(w + f) + Math.sin(ra) * Math.cos(w + f) * Math.cos(inc));
 57         ydot = ((y * h * ecc) / (r * p)) * Math.sin(f) - (h / r) * (Math.sin(ra) * Math.sin(w + f) - Math.cos(ra) * Math.cos(w + f) * Math.cos(inc));
 58         zdot = ((z * h * ecc) / (r * p)) * Math.sin(f) + (h / r) * (Math.sin(inc) * Math.cos(w + f));
 59 
 60 
 61         eciState.setX(x);
 62         eciState.setY(y);
 63         eciState.setZ(z);
 64         eciState.setVX(xdot);
 65         eciState.setVY(ydot);
 66         eciState.setVZ(zdot);
 67         eciState.setAX(0.0);
 68         eciState.setAY(0.0);
 69         eciState.setAZ(0.0);
 70         return eciState;
 71     }
 72 };
