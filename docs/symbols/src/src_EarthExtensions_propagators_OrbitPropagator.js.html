1 /*jslint browser: true, sloppy: true */
  2 /*global CoordinateConversionTools, NearCircularPropagator, RungeKuttaFehlbergPropagator */
  3 var OrbitPropagator = {
  4 
  5     /**
  6      *
  7      * @param eci                      ECICoordinates
  8      * @param elapsedTime              double
  9      * @param dt                       double, time-step
 10      * @param timeAtStartOfPropagation Date
 11      *
 12      * @returns {UNIVERSE.ECICoordinates} 
 13      */
 14     propagateOrbit: function (eci, elapsedTime, dt, timeAtStartOfPropagation) {
 15         var kep = CoordinateConversionTools.convertECIToKeplerian(eci);
 16         //console.log('eccentricity: ' + JSON.stringify(kep.getEccentricity()));
 17         //timespan is in seconds
 18         if (elapsedTime === 0.0 || isNaN(kep.getEccentricity())) {
 19             return eci;
 20         } else if (kep.getEccentricity() <= 0.1) {
 21             return NearCircularPropagator.propagateOrbit(kep, elapsedTime, dt, timeAtStartOfPropagation);
 22         } else {
 23             return RungeKuttaFehlbergPropagator.propagateOrbit(eci, elapsedTime, dt, timeAtStartOfPropagation);
 24         }
 25     }
 26 };
