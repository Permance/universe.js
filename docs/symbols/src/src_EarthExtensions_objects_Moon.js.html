1 /*jslint browser: true, sloppy: true */
  2 /*global UNIVERSE, THREE, CoordinateConversionTools, Utilities */
  3 
  4 UNIVERSE.Moon = function (universe, earthExtensions, moonImageURL) {
  5 
  6     var moonSphereSegments = 40,
  7         moonSphereRings = 30,
  8         moonSphereRadius = 1737.1,
  9 
 10         // Create the sphere
 11         geometry = new THREE.SphereGeometry(moonSphereRadius, moonSphereSegments, moonSphereRings),
 12 
 13         moonTexture = THREE.ImageUtils.loadTexture(moonImageURL),
 14 
 15         dayMaterial = new THREE.MeshPhongMaterial({
 16             map: moonTexture,
 17             color: 0xffffff,
 18             // specular: 0xffffff,
 19             //ambient: 0xffffff,
 20             // shininess: 15,
 21             //opacity: 0.5,
 22             transparent: true,
 23             // reflectivity: 1
 24             blending: THREE.AdditiveBlending
 25         }),
 26 
 27         dayMoonMesh = new THREE.Mesh(geometry, dayMaterial),
 28 
 29         moonMaterial = new THREE.MeshBasicMaterial({
 30             color: 0xffffff,
 31             overdraw: true,
 32             map: moonTexture,
 33             blending: THREE.AdditiveBlending
 34         }),
 35 
 36         moonMesh = new THREE.Mesh(geometry, moonMaterial),
 37 
 38         moonObject = new UNIVERSE.GraphicsObject(
 39             "moon",
 40             "moon",
 41             undefined,
 42             function (elapsedTime) {
 43                 var time = new Date(universe.getCurrentUniverseTime()),
 44                     propagatedValue = CoordinateConversionTools.getMoonPositionECIAtCurrentTime(time),
 45                     convertedLocation = Utilities.eciTo3DCoordinates({
 46                         x: propagatedValue.x,
 47                         y: propagatedValue.y,
 48                         z: propagatedValue.z
 49                     }, earthExtensions);
 50 
 51                 dayMoonMesh.position = {
 52                     x: convertedLocation.x,
 53                     y: convertedLocation.y,
 54                     z: convertedLocation.z
 55                 };
 56 
 57                 moonMesh.position = {
 58                     x: convertedLocation.x,
 59                     y: convertedLocation.y,
 60                     z: convertedLocation.z
 61                 };
 62                 this.currentLocation = propagatedValue;
 63             },
 64             function () {
 65                 universe.draw(this.id + "_day", dayMoonMesh, false);
 66                 universe.draw(this.id, moonMesh, false);
 67                 earthExtensions.setSunLighting(earthExtensions.useSunLighting);
 68             }
 69         );
 70     return moonObject;
 71 };
