  4 /** 
  5     A graphics object to be drawn in the Universe
  6     @constructor
  7     @param {string} id - Identifier for the object to be referenced later
  8     @param {function} updateFunction - A function(elapsedTime) that gets called each time the Universe time changes
  9     @param {function} drawFunction - A function that should call Universe.draw with the object's model
 10     @param {string} modelName - A name for the object if different than id.  Set to the id if not defined
 11     @param {Object} currentLocation
 12  */
 13 
 14 UNIVERSE.GraphicsObject = function (id, modelName, currentLocation, updateFunction, drawFunction) {
 15     this.id = id;
 16     this.modelName = modelName || id;
 17     this.currentLocation = currentLocation;
 18     this.update = updateFunction;
 19     this.draw = drawFunction;
 20 };
 21 
 22 UNIVERSE.GraphicsObject.prototype = {
 23     constructor: UNIVERSE.GraphicsObject
 24 };
