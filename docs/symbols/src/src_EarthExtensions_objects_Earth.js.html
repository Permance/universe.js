1 /*jslint browser: true, sloppy: true */
  2 /*global UNIVERSE, THREE, Constants, CoordinateConversionTools, MathTools */
  3 /** 
  4     The Earth positioned at the center of the Universe
  5     @constructor
  6     @param {UNIVERSE.Universe} universe - A Universe instance to draw the Earth in
  7     @param {UNIVERSE.EarthExtensions} earthExtensions - An EarthExtensions instance to draw the Earth with
  8     @param {URL} dayImageURL - Image to be used for the day-side of the Earth
  9     @param {URL} nightImageURL - Image to be used for the night-side of the Earth
 10  */
 11 
 12 UNIVERSE.Earth = function (universe, earthExtensions, dayImageURL, nightImageURL) {
 13     var earthSphereSegments = 40,
 14         earthSphereRings = 30,
 15 
 16     // Create the sphere
 17         geometry = new THREE.SphereGeometry(Constants.radiusEarth, earthSphereSegments, earthSphereRings),
 18         dayImageTexture   = THREE.ImageUtils.loadTexture(dayImageURL),
 19         earthAtNightTexture = THREE.ImageUtils.loadTexture(nightImageURL),
 20         nightMaterial = new THREE.MeshBasicMaterial({
 21             color: 0xffffff,
 22             overdraw: true,
 23             map: earthAtNightTexture,
 24             blending: THREE.AdditiveBlending
 25         }),
 26 
 27         nightEarthMesh = new THREE.Mesh(geometry, nightMaterial),
 28 
 29         dayMaterial = new THREE.MeshPhongMaterial({
 30             map: dayImageTexture,
 31             color: 0xffffff,
 32             // specular: 0xffffff,
 33             //ambient: 0xffffff,
 34             // shininess: 15,
 35             //opacity: 0.5,
 36             transparent: true,
 37             // reflectivity: 1
 38             blending: THREE.AdditiveBlending
 39         }),
 40 
 41         dayEarthMesh = new THREE.Mesh(geometry, dayMaterial),
 42 
 43         earthMaterial = new THREE.MeshBasicMaterial({
 44             color: 0xffffff,
 45             overdraw: true,
 46             map: dayImageTexture,
 47             blending: THREE.AdditiveBlending
 48         }),
 49 
 50         earthMesh = new THREE.Mesh(geometry, earthMaterial),
 51 
 52         previousRotation = CoordinateConversionTools.convertTimeToGMST(universe.getCurrentUniverseTime()),
 53 
 54         earthObject = new UNIVERSE.GraphicsObject(
 55             "earth",
 56             "earth",
 57             {
 58                 x: 0,
 59                 y: 0,
 60                 z: 0
 61             },
 62             function (elapsedTime) {
 63                 var rotationAngle = MathTools.toRadians(CoordinateConversionTools.convertTimeToGMST(universe.getCurrentUniverseTime()));
 64 
 65                 // TODO: This works ok with low-speed and low number of objects, not good with high speed or large number of objects
 66                 // Idea to fix it:
 67                 // Leave the camera where it is and turn on/off rotating the earth
 68                 // This will require that each will have to be converted from ECI to a rotated ECI location
 69                 // This can be buryied in the eciTo3DCoordinates method and should work so long as the math isn't overly intensive
 70                 // since it will be called A LOT
 71 
 72                 if (earthExtensions.lockCameraToWithEarthRotation) {
 73                     // move camera along with Earth
 74                     //universe.addRotationToCamera(rotationAngle - previousRotation);
 75                     earthExtensions.rotationOffsetFromXAxis += (rotationAngle - previousRotation);
 76                     if (earthExtensions.rotationOffsetFromXAxis > (2 * Math.PI)) {
 77                         earthExtensions.rotationOffsetFromXAxis -= (2 * Math.PI);
 78                     }
 79                 } else {
 80                     dayEarthMesh.rotation.y = rotationAngle - earthExtensions.rotationOffsetFromXAxis;
 81                     nightEarthMesh.rotation.y = rotationAngle - earthExtensions.rotationOffsetFromXAxis;
 82                     earthMesh.rotation.y = rotationAngle - earthExtensions.rotationOffsetFromXAxis;
 83                 }
 84 
 85                 previousRotation = rotationAngle;
 86             },
 87             function () {
 88                 // for some reason these lines have to go in this order for night to be under day...
 89                 universe.draw(this.id + "_day", dayEarthMesh, false);
 90                 universe.draw(this.id + "_night", nightEarthMesh, false);
 91                 universe.draw(this.id, earthMesh, false);
 92                 earthExtensions.setSunLighting(earthExtensions.useSunLighting);
 93 
 94             }
 95         );
 96 
 97     return earthObject;
 98 };
