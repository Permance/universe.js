  4 function KeplerianCoordinates(theSemimajorAxis, theMeanAnomaly, theEccentricAnomaly, theTrueAnomaly, theInclination, theEccentricity, theRaan, theArgOfPerigee, theMeanMotion) {
  5 
  6     this.semimajorAxis = theSemimajorAxis || 0.0; //km
  7     this.meanAnomaly = theMeanAnomaly || 0.0;
  8     this.eccentricAnomaly = theMeanAnomaly || 0.0;
  9     this.trueAnomaly = theTrueAnomaly || 0.0;
 10     this.inclination = theInclination || 0.0; //deg
 11     this.eccentricity = theEccentricity || 0.0; //unitless
 12     this.raan = theRaan || 0.0; //deg
 13     this.argOfPerigee = theArgOfPerigee || 0.0; //deg
 14     this.meanMotion = theMeanMotion || 0.0; //deg/sec
 15 
 16     this.updateAnglesUsingTrueAnomaly = function () {
 17         var nu = MathTools.toRadians(this.trueAnomaly),
 18             sinEA = Math.sin(nu) * Math.sqrt(1 - this.eccentricity * this.eccentricity) / (1 + this.eccentricity * Math.cos(nu)),
 19             cosEA = (this.eccentricity + Math.cos(nu)) / (1 + this.eccentricity * Math.cos(nu));
 20 
 21         this.EccentricAnomaly = MathTools.toDegrees(Math.atan2(sinEA, cosEA));
 22         this.MeanAnomaly = MathTools.toDegrees(MathTools.toRadians(this.EccentricAnomaly) - this.eccentricity * sinEA);
 23     };
 24 
 25     this.updateAnglesUsingMeanAnomaly = function () {
 26         /*
 27           reference vallado 2nd ed page 74 (example 2-1)
 28         */
 29 
 30         var requiredResolutionDeg = 0.00001,//deg
 31             currentError = 5000.00,//deg
 32             M = MathTools.toRadians(this.MeanAnomaly),//mean anomaly in radians
 33             Eprevious = M - this.eccentricity,
 34             Enew = 0.0,
 35             count = 0;
 36 
 37         while (currentError > requiredResolutionDeg) {
 38             Enew = Eprevious + (M - Eprevious + this.eccentricity * Math.sin(Eprevious)) / (1 - this.eccentricity * Math.cos(Eprevious));
 39             currentError = Math.abs(Enew - Eprevious);
 40             count += 1;
 41             if (count > 100) {
 42                 break;
 43             }
 44             Eprevious = Enew;
 45         }
 46         this.EccentricAnomaly = MathTools.toDegrees(Enew);
 47         this.TrueAnomaly = MathTools.toDegrees(2 * Math.atan(Math.sqrt((1 + this.eccentricity) / (1 - this.eccentricity)) * Math.tan(Enew / 2)));
 48     };
 49 
 50     this.getSemimajorAxis = function () {
 51         return this.semimajorAxis;
 52     };
 53 
 54     this.getMeanAnomaly = function () {
 55         return this.meanAnomaly;
 56     };
 57 
 58     this.getEccentricAnomaly = function () {
 59         return this.eccentricAnomaly;
 60     };
 61 
 62     this.getTrueAnomaly = function () {
 63         return this.trueAnomaly;
 64     };
 65 
 66     this.getInclination = function () {
 67         return this.inclination;
 68     };
 69 
 70     this.getEccentricity = function () {
 71         return this.eccentricity;
 72     };
 73 
 74     this.getRaan = function () {
 75         return this.raan;
 76     };
 77 
 78     this.getArgOfPerigee = function () {
 79         return this.argOfPerigee;
 80     };
 81 
 82     this.getMeanMotion = function () {
 83         return this.meanMotion;
 84     };
 85 
 86     // setters
 87     this.setSemimajorAxis = function (theSemimajorAxis) {
 88         this.semimajorAxis = theSemimajorAxis;
 89     };
 90 
 91     this.setMeanAnomaly = function (theMeanAnomaly) {
 92         this.meanAnomaly = theMeanAnomaly;
 93         this.updateAnglesUsingMeanAnomaly();
 94     };
 95 
 96     this.setEccentricAnomaly = function (theEccentricAnomaly) {
 97         this.eccentricAnomaly = theEccentricAnomaly;
 98     };
 99 
100     this.setTrueAnomaly = function (theTrueAnomaly) {
101         this.trueAnomaly = theTrueAnomaly;
102         this.updateAnglesUsingTrueAnomaly();
103     };
104 
105     this.setInclination = function (theInclination) {
106         this.inclination = theInclination;
107     };
108 
109     this.setEccentricity = function (theEccentricity) {
110         this.eccentricity = theEccentricity;
111     };
112 
113     this.setRaan = function (theRaan) {
114         this.raan = theRaan;
115     };
116 
117     this.setArgOfPerigee = function (theArgOfPerigee) {
118         this.argOfPerigee = theArgOfPerigee;
119     };
120 
121     this.setMeanMotion = function (theMeanMotion) {
122         this.meanMotion = theMeanMotion;
123     };
124 }
