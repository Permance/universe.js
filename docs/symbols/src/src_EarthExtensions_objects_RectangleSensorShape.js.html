  9 var UNIVERSE = UNIVERSE || {};

 11 UNIVERSE.RectangleSensorShape = function (shapeName, width, height) {
 12     this.shapeName = shapeName;
 13     this.width = width;
 14     this.height = height;
 15 
 16     this.getHeight = function () {
 17         return this.height;
 18     };
 19 
 20     this.setHeight = function (height) {
 21         this.height = height;
 22     };
 23 
 24     this.getWidth = function () {
 25         return this.width;
 26     };
 27 
 28     this.setWidth = function (width) {
 29         this.width = width;
 30     };
 31 
 32     this.getAngularExtentOfSensorAtSpecifiedAzimuth = function (checkAngle) {
 33 
 34         if ((this.height === 0) && (this.width === 0)) {
 35             return 0;
 36         } else {
 37             var upperRightCorner = MathTools.toDegrees(Math.atan(this.height / this.width)),  //the azimuth (from the x-axis) of the corner
 38                 upperLeftCorner = 180 - upperRightCorner,
 39                 lowerLeftCorner = 180 + upperRightCorner,
 40                 lowerRightCorner = 360 - upperRightCorner,
 41                 r1,
 42                 r2;
 43 
 44             if (checkAngle < 0) {
 45                 checkAngle += 360;
 46             }
 47 
 48             r1 = Math.abs(this.width / (2 * Math.cos(MathTools.toRadians(checkAngle))));
 49             r2 = Math.abs(this.height / (2 * Math.sin(MathTools.toRadians(checkAngle))));
 50 
 51             //determine which quadrant the azimuth is in
 52             if (checkAngle < upperRightCorner) {
 53                 return r1;
 54             } else if (checkAngle < upperLeftCorner) {
 55                 return r2;
 56             } else if (checkAngle < lowerLeftCorner) {
 57                 return r1;
 58             } else if (checkAngle < lowerRightCorner) {
 59                 return r2;
 60             } else {
 61                 return r1;
 62             }
 63 
 64         }
 65 
 66     };
 67 
 68     this.canSensorSeePointAtAzEl = function (relativeAzimuth, relativeRadius) {
 69         var canSee = false,
 70             radiusSensor = this.getAngularExtentOfSensorAtSpecifiedAzimuth(relativeAzimuth);
 71 
 72         if (radiusSensor > relativeRadius) {
 73             canSee = true;
 74         }
 75         return canSee;
 76     };
 77 };
