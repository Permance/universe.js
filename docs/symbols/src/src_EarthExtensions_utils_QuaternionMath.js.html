  5     /**
  6      *
  7      * @param q1 Quaternion
  8      * @param q2 Quaternion
  9      *
 10      * @returns Quaternion
 11      */
 12     multiplyQuaternions: function (q1, q2) {
 13         //multiplies q1 by q2;
 14         if (q1.isZero()) {
 15             return q2;
 16         } else if (q2.isZero()) {
 17             return q1;
 18         } else {
 19             var w1 = q1.getW(), //double
 20                 x1 = q1.getX(), //double
 21                 y1 = q1.getY(), //double
 22                 z1 = q1.getZ(), //double
 23                 w2 = q2.getW(), //double
 24                 x2 = q2.getX(), //double
 25                 y2 = q2.getY(), //double
 26                 z2 = q2.getZ(), //double
 27 
 28             //now that each quaternion has an axis of rotation that is a unit vector, multiply the two:
 29                 quaternionProduct = new Quaternion();
 30 
 31             quaternionProduct.setW(w1 * w2 - x1 * x2 - y1 * y2 - z1 * z2);
 32             quaternionProduct.setX(w1 * x2 + x1 * w2 + y1 * z2 - z1 * y2);
 33             quaternionProduct.setY(w1 * y2 + y1 * w2 + z1 * x2 - x1 * z2);
 34             quaternionProduct.setZ(w1 * z2 + z1 * w2 + x1 * y2 - y1 * x2);
 35 
 36             return quaternionProduct;
 37         }
 38     },

 40     /**
 41      *
 42      * @param qRotation Quaternion
 43      * @param inputVector Double[]
 44      *
 45      * @returns Double[]
 46      */
 47     applyQuaternionRotation: function (qRotation, inputVector) {
 48         //applies qRotation q1 to the vector inputVector (3x1)
 49         var q0 = qRotation.getW(), //double
 50             q1 = qRotation.getX(), //double
 51             q2 = qRotation.getY(), //double
 52             q3 = qRotation.getZ(), //double
 53 
 54             A = [], //Double[3][3];
 55             i,
 56             matrixProduct;
 57 
 58         for (i = 0; i < 3; i += 1) {
 59             A[i] = new Array(3);
 60         }
 61 
 62         A[0][0] = 2 * q0 * q0 - 1 + 2 * q1 * q1;
 63         A[0][1] = 2 * q1 * q2 + 2 * q0 * q3;
 64         A[0][2] = 2 * q1 * q3 - 2 * q0 * q2;
 65         A[1][0] = 2 * q1 * q2 - 2 * q0 * q3;
 66         A[1][1] = 2 * q0 * q0 - 1 + 2 * q2 * q2;
 67         A[1][2] = 2 * q2 * q3 + 2 * q0 * q1;
 68         A[2][0] = 2 * q1 * q3 + 2 * q0 * q2;
 69         A[2][1] = 2 * q2 * q3 - 2 * q0 * q1;
 70         A[2][2] = 2 * q0 * q0 - 1 + 2 * q3 * q3;
 71 
 72         matrixProduct = MathTools.multiply2dBy1d(A, [inputVector.x, inputVector.y, inputVector.z]);
 73 
 74         return new THREE.Vector3(
 75             matrixProduct[0],
 76             matrixProduct[1],
 77             matrixProduct[2]
 78         );
 79     },

 81     /**
 82      *
 83      * @param q Quaternion
 84      *
 85      * @returns Double[]
 86      */
 87     getEulerAngles: function (q) {
 88         //------DETERMINING THE EQUIVALENT EULER ROTATION ANGLES------
 89         //translate the net (final) quaternion back to euler angles
 90         var q0 = q.getW(), //double
 91             q1 = q.getX(), //double
 92             q2 = q.getY(), //double
 93             q3 = q.getZ(), //double
 94 
 95             phi = Math.atan2((2 * (q0 * q1 + q2 * q3)),
 96                 (1 - 2 * (q1 * q1 + q2 * q2))),   //rad  (rotation about the x-axis)
 97             theta = Math.asin(2 * (q0 * q2 - q3 * q1)), //rad  (rotation about the y-axis)
 98             gamma = Math.atan2((2 * (q0 * q3 + q1 * q2)),
 99                 (1 - 2 * (q2 * q2 + q3 * q3))), //rad  (rotation about the z-axis)
100 
101         //equivalentRotationMatrix=Rot3(gamma*180/pi)*Rot2(theta*180/pi)*Rot1(phi*180/pi);
102             EulerAngles = []; //Double[3];
103 
104         EulerAngles[0] = MathTools.toDegrees(phi);     //deg  (rotation about the x-axis)
105         EulerAngles[1] = MathTools.toDegrees(theta);   //deg  (rotation about the y-axis)
106         EulerAngles[2] = MathTools.toDegrees(gamma);   //deg  (rotation about the z-axis)
107 
108         //remember, the equivalentRotationMatrix=Rot3(gamma*180/pi)*Rot2(theta*180/pi)*Rot1(phi*180/pi);
109         return EulerAngles;
110     },

118     convertRotationMatrixToQuaternion: function (m) {
119         //converts a 3x3 rotation matrix to an equivalent quaternion
120         var q = new Quaternion(),
121             m11 = m[0][0], //double
122             m12 = m[0][1], //double
123             m13 = m[0][2], //double
124             m21 = m[1][0], //double
125             m22 = m[1][1], //double
126             m23 = m[1][2], //double
127             m31 = m[2][0], //double
128             m32 = m[2][1], //double
129             m33 = m[2][2]; //double
130 
131         // q.setW(0.5 * Math.sqrt(m11+m22+m33+1));
132         //         q.setX((m23 - m32) / (4 * q.getW()));
133         //         q.setY((m31 - m13) / (4 * q.getW()));
134         //         q.setZ((m12 - m21) / (4 * q.getW()));
135 
136         q.setW(0.5 * Math.sqrt(1 + m11 - m22 - m33));
137         q.setX((m12 + m21) / (4 * q.getW()));
138         q.setY((m13 + m31) / (4 * q.getW()));
139         q.setZ((m32 - m23) / (4 * q.getW()));
140 
141         return q;
