 16     //default the sensor to orient along the satellite's nadir (it defaults to align with RSW where R would be equivalent to the sensor's centerline)

 18     this.rotationRadians = MathTools.toRadians(180.0);

 20     // http://www.cprogramming.com/tutorial/3d/quaternions.html
 21     this.rotationAxis = {
 22         x: 0.0,
 23         y: 0.0,
 24         z: 1.0
 25     };
 26 
 27     this.quaternionFromRSWToSensor.setW(Math.cos(this.rotationRadians / 2.0));
 28     this.quaternionFromRSWToSensor.setX(Math.sin(this.rotationRadians / 2.0) * this.rotationAxis.x);
 29     this.quaternionFromRSWToSensor.setY(Math.sin(this.rotationRadians / 2.0) * this.rotationAxis.y);
 30     this.quaternionFromRSWToSensor.setZ(Math.sin(this.rotationRadians / 2.0) * this.rotationAxis.z);

 34         if (rotationAngle && rotationAngle !== 0) {

 37                 rotationAxis = {

 39                     y: 0.0,

 42

117     this.determineTargetAzElRelativeToSensor = function (satellite, targetPosition) {
118         //define the target position as a vectors
119         //console.log("satellite: " + satellite.getEci().getX() + ", " + satellite.getEci().getY() + ", " + satellite.getEci().getZ() +  '      ' + 
120         //            "targetpos: " + targetPosition.getX() + ", " + targetPosition.getY() + ", " + targetPosition.getZ());
121         //System.out.println("targetpos: " + targetPosition.getX() + ", " + targetPosition.getY() + ", " + targetPosition.getZ());
122 
123         var satelliteEci = satellite.getEci(),
124             deltaECI = new THREE.Vector3(
125                 targetPosition.x - satelliteEci.getX(),
126                 targetPosition.y - satelliteEci.getY(),
127                 targetPosition.z - satelliteEci.getZ()
128             ),
129             r = new THREE.Vector3(
130                 satelliteEci.getX(),
131                 satelliteEci.getY(),
132                 satelliteEci.getZ()
133             ),
134             v = new THREE.Vector3(
135                 satelliteEci.getVX(),
136                 satelliteEci.getVY(),
137                 satelliteEci.getVZ()
138             ),
139             rmag = r.length(),
140             rcrossv = new THREE.Vector3(),
141             rvec = new THREE.Vector3(),
142             w = new THREE.Vector3(),
143             s = new THREE.Vector3(),
144             deltaECIdotR,
145             deltaECIdotS,
146             deltaECIdotW,
147             targetInRSWCoordinates,
148             targetInSensorCoordinates,
149             centerline,
150             rightline,
151             topline,
152             a,
153             b,
154             c,
155             d,
156             e,
157             S,
158             C,
159             az,
160             el;
161 
162         rcrossv.cross(r, v);
163 
164         rvec.copy(r);
165         rvec.divideScalar(rmag);
166 
167         w.copy(rcrossv);
168         w.divideScalar(rcrossv.length());
169 
170         s.cross(w, r);
171 
172         deltaECIdotR = deltaECI.dot(rvec);
173 
174         deltaECIdotS = deltaECI.dot(s);
175 
176         deltaECIdotW = deltaECI.dot(w);
177 
178         targetInRSWCoordinates = new THREE.Vector3(
179             deltaECIdotR / rvec.length(),
180             deltaECIdotS / s.length(),
181             deltaECIdotW / w.length()
182         );
183 
184         //get the quaternion to convert the ECI coordinate of the target into an RSW coordinate triplet
185         targetInSensorCoordinates = QuaternionMath.applyQuaternionRotation(this.quaternionFromRSWToSensor, targetInRSWCoordinates);
186 
187         //---determine the three vectors that define the extent of the sensor shape in the sensor coordinate system
188         //these are the non-rotated vectors
189         //vector along the centerline of the sensor (RSW)
190         centerline = new THREE.Vector3(
191             1.0,//center (radial)
192             0.0,//left
193             0.0//top
194         );
195 
196         rightline = new THREE.Vector3(
197             1.0,//center
198             -Math.tan(MathTools.toRadians(this.shape.getAngularExtentOfSensorAtSpecifiedAzimuth(0.0))),//left
199             0.0//top
200         );
201 
202         topline = new THREE.Vector3(
203             1.0,//center
204             0.0,//left
205             Math.tan(MathTools.toRadians(this.shape.getAngularExtentOfSensorAtSpecifiedAzimuth(90.0)))//top
206         );
207 
208         //calculate the azimuth and elevation angles of the target relative to the centerline of the sensor FOV
209         //assume it's an oblique spherical triangle
210         //reference http://mathworld.wolfram.com/SphericalTrigonometry.html
211         //a is the angle between the centerline and the right hand side
212         a = MathTools.toRadians(MathTools.angleBetweenTwoVectorsVector(centerline, rightline));//radians
213 
214         //b is the angle between the centerline and the target
215         b = MathTools.toRadians(MathTools.angleBetweenTwoVectorsVector(centerline, targetInSensorCoordinates));//radians
216 
217         //c is the angle between the right hand side and the target
218         c = MathTools.toRadians(MathTools.angleBetweenTwoVectorsVector(rightline, targetInSensorCoordinates));//radians
219 
220         //d is the angle between the top side and the target
221         d = MathTools.toRadians(MathTools.angleBetweenTwoVectorsVector(topline, targetInSensorCoordinates));//radians
222 
223         //e is the angle between the top side and the target
224         e = MathTools.toRadians(MathTools.angleBetweenTwoVectorsVector(topline, centerline));//radians
225 
226         S = 0.5 * (a + b + c);
227         C = 2 * Math.asin(Math.sqrt((Math.sin(Math.abs(S - a)) * Math.sin(Math.abs(S - b)) / (Math.sin(a) * Math.sin(b)))));
228 
229         //double C=Math.acos(Math.cos(c)-Math.cos(a)*Math.cos(b))/(Math.sin(a)*Math.sin(b));//radians
230         az = MathTools.toDegrees(C);
231         el = MathTools.toDegrees(b);
232 
233         if (d > b && d > e) {
234             az = 360 - az;
235         }
236         //correct the azimuth
237         //System.out.println("a (CL-RL): " + MathTools.toDegrees(a) + " b (CL-TGTL): " + MathTools.toDegrees(b) + " c (RL-TGTL): " + MathTools.toDegrees(c) + " d (TL-TGTL): " + MathTools.toDegrees(d) + " e (TL-CL): " + MathTools.toDegrees(e) + " C (az):" + MathTools.toDegrees(C));
238 
239 
240         //return the azimuth and elevation
241         return {
242             az: az,
243             el: el
244         };
245     };

247     this.checkToSeeIfEarthObscuresLineBetweenSatelliteAndTargetSphericalEarth = function (satellite, targetPosition) {
248         //sight algorithm Vallado 295
249         var haveSight = false,
250             radEarth = Constants.radiusEarth,
251             satelliteEci = satellite.getEci(),
252             r1 = new THREE.Vector3(
253                 satelliteEci.getX() / radEarth,
254                 satelliteEci.getY() / radEarth,
255                 satelliteEci.getZ() / radEarth
256             ),
257             r2 = new THREE.Vector3(
258                 targetPosition.getX() / radEarth,
259                 targetPosition.getY() / radEarth,
260                 targetPosition.getZ() / radEarth
261             ),
262 
263             r1mag = r1.length(),
264             r2mag = r2.length(),
265             tmin = 0.5,
266             ctmin = 0.0,
267             r1DotR2;
268 
269         if (r1mag < 1 || r2mag < 1) { //one of the points is inside the earth
270             haveSight = false;
271         } else {
272             r1DotR2 = r1.dot(r2);
273             tmin = (r1mag * r1mag - r1DotR2) / (r1mag * r1mag + r2mag * r2mag - 2 * r1DotR2);
274             if (tmin < 0 || tmin > 1) {
275                 haveSight = true;
276             } else {
277                 ctmin = (1 - tmin) * r1mag * r1mag + r1DotR2 * tmin;
278                 if (ctmin >= 1.0) {
279                     haveSight = true;
280                 }
281             }
282 
283         }
284         return !haveSight;
285     };

332     this.checkSensorVisibilityOfTargetPoint = function (satellite, targetPosition) {
333         //first, correct the target azimuth and elevation to be in the same frame of reference as the sensor FOV
334         var azel = this.determineTargetAzElRelativeToSensor(satellite, targetPosition),
335 
336         //then check to see if this point is in the field of view of the sensor
337             inFOV = this.shape.canSensorSeePointAtAzEl(azel.az, azel.el),
338             earthObscured = this.checkToSeeIfEarthObscuresLineBetweenSatelliteAndTargetOblateEarth(satellite, targetPosition);
339 
340         //console.log("earth obscured:" + earthObscured + "    inFOV:" + inFOV);
341         if (earthObscured) {
342             return false;
343         } else {
344             if (inFOV === true && earthObscured === false) { //if it's in the field of view and not obscured, indicate that you can see the target
345                 return true;
346             } else {
347                 return false;
348             }
349         }

352     this.buildPointsToDefineSensorShapeInECI = function (NumPoints, satellite) {
353         //returns the instantaneous field of view of the sensor as represented by 
354         //a set of 'NumPoints' unit vectors from the center of the satellite to the boundaries
355         //of the sensor field of view in ECI coordinates
356         //it is assumed that each xyz pair 
357         var FOV = [],
358             azimuthStep = 360.0 / NumPoints, //the azimuth separation between each point
359             i,
360             az,
361             el,
362             FOVboundary,
363             FOVmagnitude,
364             rswPoint,
365             satelliteEci,
366             satXpos,
367             satYpos,
368             satZpos,
369             eciTemp;
370 
371         for (i = 0; i < NumPoints; i += 1) {
372             az = i * azimuthStep;
373             el = MathTools.toRadians(this.shape.getAngularExtentOfSensorAtSpecifiedAzimuth(az));
374             //build the sensor field of view vector in RSW
375 
376             //define the vector to the sensor boundary
377             FOVboundary = new THREE.Vector3(
378                 1.0, // radial
379                 el * Math.cos(MathTools.toRadians(az)), //along
380                 el * Math.sin(MathTools.toRadians(az))  //cross
381             );
382 
383             //ensure that it is a unit vector
384             FOVmagnitude = FOVboundary.length();
385 
386             FOVboundary = FOVboundary.divideScalar(FOVmagnitude);
387 
388             //console.log("FOVboundary1: " + JSON.stringify(FOVboundary));
389 
390             FOVboundary = QuaternionMath.applyQuaternionRotation(this.quaternionFromRSWToSensor, FOVboundary);
391 
392             //console.log("FOVboundary2: " + JSON.stringify(FOVboundary));
393 
394             rswPoint = new UNIVERSE.RSWCoordinates(FOVboundary.x, FOVboundary.y, FOVboundary.z);
395 
396             satelliteEci = satellite.getEci();
397             satXpos = satelliteEci.getX();
398             satYpos = satelliteEci.getY();
399             satZpos = satelliteEci.getZ();
400 
401             //convert the RSW to ECI
402             eciTemp = CoordinateConversionTools.convertRSWToECI(satellite, rswPoint);
403             FOV[i] = new THREE.Vector3(
404                 satXpos + eciTemp.getX(),
405                 satYpos + eciTemp.getY(),
406                 satZpos + eciTemp.getZ()
407             );
408         }
409 
410         return FOV;
411     };

413     this.findProjectionPoints = function (endpoints, satellite, distancePastEarthToDraw) {
414         var satellitePositionTemp = satellite.getEci(),
415             satellitePosition = new THREE.Vector3(satellitePositionTemp.x, satellitePositionTemp.y, satellitePositionTemp.z),
416 
417             shiftedEarthCenter = new THREE.Vector3(
418                 -satellitePosition.x,
419                 -satellitePosition.y,
420                 -satellitePosition.z
421             ),
422 
423             pointsOnEarth = [],
424 
425             endPointLen = endpoints.length,
426             i,
427             shiftedBoundaryPoint,
428             satelliteDistanceFromCenterOfEarth,
429             depth,
430             I,
431             c,
432             aboveTheEarth = 100,
433             r = Constants.radiusEarth + aboveTheEarth, // r is the radius above the earth to project the points....this adds an arbitrary number above the earth to minimize drawing collisions
434             Idotc,
435             cdotc,
436             valueToBeSqrtd,
437             sqrtValue,
438             distanceToIntersectionPlus,
439             distanceToIntersectionMinus,
440             pointOnEarth,
441             extendedEndPoint,
442             unitShiftedEarthCenter,
443             angleBetweenEarthCenterAndExtendedEndPoint,
444             height,
445             distanceToTangentPoint,
446             scaleFromEndPointToTangentPoint,
447             tangentPoint,
448             tangentPointMagnitude,
449             scaleToEarthSurface,
450             tangentPointOnSurface;
451 
452         for (i = 0; i < endPointLen; i += 1) {
453             // shift everything to the satellite position as the origin
454             shiftedBoundaryPoint = new THREE.Vector3(
455                 endpoints[i].x - satellitePosition.x,
456                 endpoints[i].y - satellitePosition.y,
457                 endpoints[i].z - satellitePosition.z
458             );
459 
460             satelliteDistanceFromCenterOfEarth = satellitePosition.length();
461             depth = satelliteDistanceFromCenterOfEarth + distancePastEarthToDraw;
462 
463             // Based on this: http://en.wikipedia.org/wiki/Line%E2%80%93sphere_intersection
464             // d = ((I*c) +/- sqrt((I*c)^2 - c*c + r*r))
465             I = shiftedBoundaryPoint;
466             c = shiftedEarthCenter;
467 
468             Idotc = I.dot(c);
469 
470             cdotc = c.dot(c);
471 
472             valueToBeSqrtd = Idotc * Idotc - cdotc + r * r;
473 
474             // Line does intersect the earth so find the point
475             if (valueToBeSqrtd >= 0) {
476                 sqrtValue = Math.sqrt(Idotc * Idotc - cdotc + r * r);
477                 distanceToIntersectionPlus = Idotc + sqrtValue;
478                 distanceToIntersectionMinus = Idotc - sqrtValue;
479 
480                 pointOnEarth = new THREE.Vector3(
481                     shiftedBoundaryPoint.x * distanceToIntersectionMinus + satellitePosition.x,
482                     shiftedBoundaryPoint.y * distanceToIntersectionMinus + satellitePosition.y,
483                     shiftedBoundaryPoint.z * distanceToIntersectionMinus + satellitePosition.z
484                 );
485                 if (pointOnEarth.x) {
486                     pointsOnEarth.push(pointOnEarth);
487                 }
488 
489             } else { // This means the line does not intersect the earth and we will find the nearest tangent point
490                 // Scale the boundary point out to the depth
491                 extendedEndPoint = new THREE.Vector3();
492                 extendedEndPoint.copy(shiftedBoundaryPoint);
493                 extendedEndPoint.multiplyScalar(depth);
494 
495                 // Get a unit vector pointed at the earth center from the satellite coordinate system
496                 unitShiftedEarthCenter = new THREE.Vector3();
497                 unitShiftedEarthCenter.copy(shiftedEarthCenter);
498                 unitShiftedEarthCenter.divideScalar(satelliteDistanceFromCenterOfEarth);
499 
500                 // Find the angle between the two
501                 angleBetweenEarthCenterAndExtendedEndPoint = Math.acos(shiftedBoundaryPoint.dot(unitShiftedEarthCenter));
502 
503                 // http://www.algebralab.org/studyaids/studyaid.aspx?file=Trigonometry_LawSines.xml
504                 height = satelliteDistanceFromCenterOfEarth * Math.sin(angleBetweenEarthCenterAndExtendedEndPoint);
505 
506                 // Distance along the line where the tangent point is.  i.e. where the height touches the line
507                 distanceToTangentPoint = height / Math.atan(angleBetweenEarthCenterAndExtendedEndPoint);
508 
509                 // Scale to the point and then shift back to earth-centered
510                 scaleFromEndPointToTangentPoint = distanceToTangentPoint / depth;
511                 tangentPoint = new THREE.Vector3(
512                     (extendedEndPoint.x) * scaleFromEndPointToTangentPoint + satellitePosition.x,
513                     (extendedEndPoint.y) * scaleFromEndPointToTangentPoint + satellitePosition.y,
514                     (extendedEndPoint.z) * scaleFromEndPointToTangentPoint + satellitePosition.z
515                 );
516 
517                 // scale down to the earth's surface, plus an arbitrary distance above the earth
518                 tangentPointMagnitude = tangentPoint.length();
519                 scaleToEarthSurface = (Constants.radiusEarth + aboveTheEarth) / tangentPointMagnitude;
520 
521                 tangentPointOnSurface = new THREE.Vector3();
522                 tangentPointOnSurface.copy(tangentPoint);
523                 tangentPointOnSurface.multiplyScalar(scaleToEarthSurface);
524 
525                 pointsOnEarth.push(tangentPointOnSurface);
526             }
527 
528         }
529 
530         return pointsOnEarth;
531     };
