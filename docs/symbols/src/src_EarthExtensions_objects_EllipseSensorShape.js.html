  2 /*global MathTools */
  3 // EllipseSensorShape.js
  4 
  5 var UNIVERSE = UNIVERSE || {};
  6 
  7 /** 
  8     Represents an Ellipse sensor shape to be used in Sensor projection and visibility calculation
  9     @constructor
 10     @param {string} shapeName - Name of the sensor shape
 11     @param {double} semiMajorAngle - SMA of the ellipse sensor
 12     @param {double} semiMinorAngle - SMI of the elipse sensor
 13  */
 14 UNIVERSE.EllipseSensorShape = function (shapeName, semiMajorAngle, semiMinorAngle) {
 15     this.shapeName = shapeName;
 16     this.semiMajorAngle = semiMajorAngle;
 17     this.semiMinorAngle = semiMinorAngle;
 18 
 19     this.getSemiMajorAngle = function () {
 20         return this.semiMajorAngle;
 21     };
 22 
 23     this.setSemiMajorAngle = function (semiMajorAngle) {
 24         this.semiMajorAngle = semiMajorAngle;
 25     };
 26 
 27     this.getSemiMinorAngle = function () {
 28         return this.semiMinorAngle;
 29     };
 30 
 31     this.setSemiMinorAngle = function (semiMinorAngle) {
 32         this.semiMinorAngle = semiMinorAngle;
 33     };
 34 
 35     this.getAngularExtentOfSensorAtSpecifiedAzimuth = function (checkAngle) {
 36         if ((this.semiMajorAngle === 0) && (this.semiMinorAngle === 0)) {
 37             return 0;
 38         } else {
 39             var a = MathTools.toRadians(this.semiMajorAngle),
 40                 b = MathTools.toRadians(this.semiMinorAngle),
 41                 cosTheta = Math.cos(MathTools.toRadians(checkAngle)),
 42                 sinTheta = Math.sin(MathTools.toRadians(checkAngle));
 43             return MathTools.toDegrees(a * b / Math.sqrt((b * cosTheta) * (b * cosTheta) + (a * sinTheta) * (a * sinTheta)));
 44         }
 45     };
 46 
 47     this.canSensorSeePointAtAzEl = function (relativeAzimuth, relativeRadius) {
 48         var canSee = false,
 49             radiusSensor = this.getAngularExtentOfSensorAtSpecifiedAzimuth(relativeAzimuth);
 50 
 51         if (radiusSensor > relativeRadius) {
 52             canSee = true;
 53         }
 54         return canSee;
 55     };
 56 };
