  4 /** 
  5     An object to be drawn in orbit around the Earth
  6     @constructor
  7     @param {string} id - Identifier for the object to be referenced later
  8     @param {string} objectName - A name for the object if different than id.  Set to the id if not defined
  9     @param {function} propagator - A function(time) to give the object's position at a time.  No time passed in means the current Universe time
 10     @param {string} modelId - Identifier for the model to use that has been added to the Universe's object library
 11     @param {boolean} showPropagationLine - should a propagation line be shown for the object
 12     @param {boolean} showGroundTrackPoint - should the ground track point be shown for the object
 13  */
 14 
 15 UNIVERSE.SpaceObject = function (id, objectName, modelId, propagator, showPropagationLine, showGroundTrackPoint, sensors, currentLocation, universe, earthExtensions) {
 16     if (!id) {
 17         return undefined;
 18     }
 19     this.id = id;
 20     this.objectName = objectName || id;
 21     this.propagator = propagator;
 22     this.modelId = modelId;
 23     this.showPropagationLine = showPropagationLine || false;
 24     this.showGroundTrackPoint = showGroundTrackPoint || false;
 25     this.sensors = sensors || undefined;
 26 
 27     this.currentLocation = currentLocation || undefined;
 28     this.universe = universe || undefined;
 29     this.earthExtensions = earthExtensions || undefined;
 30 };
 31 
 32 UNIVERSE.SpaceObject.prototype = {
 33     constructor: UNIVERSE.SpaceObject,
 34 
 35     set: function (id, objectName, propagator, modelId, showPropogationLine, showGroundTrackPoint, sensors, currentLocation, universe, earthExtensions) {
 36 
 37         this.id = id;
 38         this.objectName = objectName || id;
 39         this.propagator = propagator;
 40         this.modelId = modelId;
 41         this.showPropagationLine = showPropogationLine || false;
 42         this.showGroundTrackPoint = showGroundTrackPoint || false;
 43         this.sensors = sensors || undefined;
 44         this.currentLocation = currentLocation || undefined;
 45         this.universe = universe || undefined;
 46         this.earthExtensions = earthExtensions || undefined;
 47 
 48         return this;
 49     },
 50 
 51     getEci: function () {
 52         //var location = this.propagator();
 53         return this.currentLocation;
 54     },
 55 
 56     getGraphicsObject: function (material, objectGeometry, universe, earthExtensions) {
 57 
 58         this.universe = universe;
 59         this.earthExtensions = earthExtensions;
 60         var spaceObject = this,
 61             objectModel,
 62             spaceGraphicsObject;
 63 
 64         objectGeometry.applyMatrix(new THREE.Matrix4().setRotationFromEuler(new THREE.Vector3(-Math.PI / 2, 0, 0)));
 65         objectModel = new THREE.Mesh(objectGeometry, material);
 66 
 67         spaceGraphicsObject = new UNIVERSE.GraphicsObject(
 68             this.id,
 69             this.objectName,
 70             undefined,
 71             function (elapsedTime) {
 72                 // need to pass a time to the propagator
 73                 var propagatedLocation = spaceObject.propagator(),
 74                     convertedLocation = Utilities.eciTo3DCoordinates(propagatedLocation, earthExtensions);
 75                 if (convertedLocation) {
 76                     objectModel.position.set(convertedLocation.x, convertedLocation.y, convertedLocation.z);
 77 
 78                     //http://mrdoob.github.com/three.js/examples/misc_lookat.html
 79                     objectModel.lookAt(new THREE.Vector3(0, 0, 0));
 80                     this.currentLocation = propagatedLocation;
 81                     //console.log("currentLocation: " + spaceObject.currentLocation);
 82                 }
 83             },
 84             function () {
 85                 spaceObject.universe.draw(this.id, objectModel, false);
 86                 spaceObject.earthExtensions.showModelForId(spaceObject.showVehicle, this.id);
 87             }
 88         );
 89         return spaceGraphicsObject;
 90     }
 91 };
