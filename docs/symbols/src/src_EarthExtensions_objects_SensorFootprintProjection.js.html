  6 /**

  8  */
  9 var UNIVERSE = UNIVERSE || {};
 10 
 11 UNIVERSE.SensorFootprintProjection = function (sensor, object, universe, earthExtensions, objectLocation) {
 12 
 13     var objectMaterial = new THREE.LineBasicMaterial({
 14         color : Utilities.get_random_color(),
 15         opacity : 0.5,
 16         linewidth : 3
 17     }),
 18         objectGeometry = new THREE.Geometry(),
 19 
 20         points = sensor.buildPointsToDefineSensorShapeInECI(40, object),
 21     //var extendedPoints = sensors[0].extendSensorEndpointsInECIToConformToEarth(points, spaceObject, 1000, 10);
 22         extendedPoints = sensor.findProjectionPoints(points, object, 1000),
 23         j,
 24         vector,
 25         line,
 26         lineGraphicsObject;
 27 
 28     for (j = 0; j < extendedPoints.length; j += 1) {
 29         vector = new THREE.Vector3(-extendedPoints[j].x, extendedPoints[j].z, extendedPoints[j].y);
 30         objectGeometry.vertices.push(new THREE.Vertex(vector));
 31     }
 32 
 33     objectGeometry.vertices.push(new THREE.Vertex(new THREE.Vector3(-extendedPoints[0].x, extendedPoints[0].z, extendedPoints[0].y)));
 34 
 35     line = new THREE.Line(objectGeometry, objectMaterial);
 36  
 37     lineGraphicsObject = new UNIVERSE.GraphicsObject(
 38         object.id + "_footprint_" + sensor.name,
 39         undefined,
 40         undefined,
 41         function (elapsedTime) {
 42             if (earthExtensions.enableSensorFootprintProjections) {
 43                 var points = this.sensor.buildPointsToDefineSensorShapeInECI(40, object),
 44                     extendedPoints = this.sensor.findProjectionPoints(points, object, 1000),
 45                     k,
 46                     convertedLocation,
 47                     convertedLastPoint;
 48 
 49                 for (k = 0; k < extendedPoints.length; k += 1) {
 50                     convertedLocation = Utilities.eciTo3DCoordinates(extendedPoints[k], earthExtensions);
 51                     line.geometry.vertices[k].position = {
 52                         x: convertedLocation.x,
 53                         y: convertedLocation.y,
 54                         z: convertedLocation.z
 55                     };
 56                 }
 57 
 58                 convertedLastPoint = Utilities.eciTo3DCoordinates(extendedPoints[0], earthExtensions);
 59                 line.geometry.vertices[extendedPoints.length].position = {
 60                     x: convertedLastPoint.x,
 61                     y: convertedLastPoint.y,
 62                     z: convertedLastPoint.z
 63                 };
 64 
 65                 line.geometry.__dirtyVertices = true;
 66             }
 67         },
 68         function () {
 69             universe.draw(this.id, line, false);
 70         }
 71     );
 72     lineGraphicsObject.sensor = sensor;
 73 
 74     return lineGraphicsObject;
 75 };
