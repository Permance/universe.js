  5     /**
  6      *
  7      *
  8      * @param state                    double[]
  9      * @param elapsedTime              double
 10      * @param dt                       double
 11      * @param timeAtStartOfPropagation date
 12      *
 13      * @returns Array of doubles
 14      */
 15     rungeKuttaFehlbergIntegrator: function (state, elapsedTime, dt, timeAtStartOfPropagation) {
 16         //fifth order runge-kutta-fehlberg integrator
 17         var tempState = state,
 18 
 19             f = [],          //array of doubles[9]
 20             deltaState = [], //array of doubles[9]
 21             dtPrime = 0.0,            //double
 22             N = 1,  //int
 23             h = dt, //double
 24 
 25             //if the timestep is too big, reduce it to a smaller timestep and loop through the updates at the smaller timestep to add up to the total timestep
 26             maxResolution = 0.02, //double
 27             newMilliseconds = timeAtStartOfPropagation.getTime(), //long
 28             j,
 29             simTime,
 30             k1,
 31             k2,
 32             k3,
 33             k4,
 34             k5,
 35             k6,
 36             i,
 37             GST;
 38 
 39         if (dt > maxResolution) {
 40             h = maxResolution;
 41         }
 42 
 43         N = elapsedTime / h;
 44 
 45         for (j = 1; j <= N; j += 1) {
 46             newMilliseconds = newMilliseconds + (h * 1000.0); //keep in millis
 47             simTime = new Date(newMilliseconds);
 48             GST = CoordinateConversionTools.convertTimeToGMST(simTime); //double
 49 
 50             k1 = []; //double[9]
 51             k2 = []; //double[9]
 52             k3 = []; //double[9]
 53             k4 = []; //double[9]
 54             k5 = []; //double[9]
 55             k6 = []; //double[9]
 56 
 57             //build k1
 58             for (i = 0; i < 9; i += 1) {
 59                 deltaState[i] = tempState[i];
 60             }
 61 
 62             dtPrime = h;
 63 
 64             f = this.generateStateUpdate(deltaState, dtPrime, GST);
 65 
 66             for (i = 0; i < 9; i += 1) {
 67                 k1[i] = h * f[i];
 68             }
 69 
 70             //build k2
 71             for (i = 0; i < 9; i += 1) {
 72                 deltaState[i] = tempState[i] + 0.25 * k1[i];
 73             }
 74 
 75             dtPrime = 0.25 * h;
 76 
 77             f = this.generateStateUpdate(deltaState, dtPrime, GST);
 78 
 79             for (i = 0; i < 9; i += 1) {
 80                 k2[i] = h * f[i];
 81             }
 82 
 83             //build k3
 84             for (i = 0; i < 9; i += 1) {
 85                 deltaState[i] = tempState[i] + (3.0 / 32.0) * k1[i] + (9.0 / 32.0) * k2[i];
 86             }
 87 
 88             dtPrime = 0.375 * h;
 89 
 90             f = this.generateStateUpdate(deltaState, dtPrime, GST);
 91 
 92             for (i = 0; i < 9; i += 1) {
 93                 k3[i] = h * f[i];
 94             }
 95 
 96             //build k4
 97             for (i = 0; i < 9; i += 1) {
 98                 deltaState[i] = tempState[i] + ((1932.0 / 2197.0) * k1[i]) -
 99                     ((7200.0 / 2197.0) * k2[i]) + ((7296.0 / 2197.0) * k3[i]);
100             }
101 
102             dtPrime = 0.9230769230769231 * h;
103 
104             f = this.generateStateUpdate(deltaState, dtPrime, GST);
105 
106             for (i = 0; i < 9; i += 1) {
107                 k4[i] = h * f[i];
108             }
109 
110             //build k5
111             for (i = 0; i < 9; i += 1) {
112                 deltaState[i] = tempState[i] + ((439.0 / 216.0) * k1[i]) -
113                     (8.0 * k2[i]) + ((3680.0 / 513.0) * k3[i]) - ((845.0 / 4104.0) * k4[i]);
114             }
115 
116             dtPrime = h;
117 
118             f = this.generateStateUpdate(deltaState, dtPrime, GST);
119 
120             for (i = 0; i < 9; i += 1) {
121                 k5[i] = h * f[i];
122             }
123 
124             //build k6
125             for (i = 0; i < 9; i += 1) {
126                 deltaState[i] = tempState[i] - ((8.0 / 27.0) * k1[i]) +
127                     ((2.0) * k2[i]) - ((3544.0 / 2565.0) * k3[i]) +
128                     ((1859.0 / 4104.0) * k4[i]) - ((11.0 / 40.0) * k5[i]);
129             }
130 
131             dtPrime = (0.5) * h;
132 
133             f = this.generateStateUpdate(deltaState, dtPrime, GST);
134 
135             for (i = 0; i < 9; i += 1) {
136                 k6[i] = h * f[i];
137             }
138 
139             //generate the estimate for this step in time
140             for (i = 0; i < 9; i += 1) {
141                 //http://en.wikipedia.org/wiki/Runge%E2%80%93Kutta%E2%80%93Fehlberg_method
142                 tempState[i] = tempState[i] + ((16.0 / 135.0) * k1[i]) +
143                     ((6656.0 / 12825.0) * k3[i]) + ((28561.0 / 56430.0) * k4[i]) -
144                     ((9.0 / 50.0) * k5[i]) + ((2.0 / 55.0) * k6[i]);
145             }
146         }
147 
148         return tempState;
149     },
150 
151     /**
152      *
153      *
154      * @param state double[]
155      * @param dt    double
156      * @param GST   double
157      *
158      * @returns double[]
159      */
160     generateStateUpdate: function (state, dt, GST) {
161         //state is 9x1
162         //state structure is x,y,z,vx,vy,vz,ax,ay,az
163         var stateRateOfChange = [], //double[9]
164             mu = Constants.muEarth,     //double
165             r = Math.sqrt((state[0] * state[0]) + (state[1] * state[1]) +
166                 (state[2] * state[2])); //double
167 
168         //figure out the rate of change of x,y,z,vx,vy,vz
169         stateRateOfChange[0] = state[3]; //vx
170         stateRateOfChange[1] = state[4]; //vy
171         stateRateOfChange[2] = state[5]; //vz
172         stateRateOfChange[3] = -mu * state[0] / (r * r * r); //ax
173         stateRateOfChange[4] = -mu * state[1] / (r * r * r); //ay
174         stateRateOfChange[5] = -mu * state[2] / (r * r * r); //az
175         stateRateOfChange[6] = 0.0;
176         stateRateOfChange[7] = 0.0;
177         stateRateOfChange[8] = 0.0;
178 
179         return stateRateOfChange;
180     },

182     propagateOrbit: function (eci, elapsedTime, dt, timeAtStartOfPropagation) {
183         var state = [],
184             updatedState,
185             newEci;     //double[9]
186 
187         //establish the starting state vector;
188         state[0] = eci.x;
189         state[1] = eci.y;
190         state[2] = eci.z;
191         state[3] = eci.vx;
192         state[4] = eci.vy;
193         state[5] = eci.vz;
194         state[6] = eci.ax;
195         state[7] = eci.ay;
196         state[8] = eci.az;
197 
198         //call the integrator
199         updatedState = this.rungeKuttaFehlbergIntegrator(state, elapsedTime, dt, timeAtStartOfPropagation);
200 
201         //translate the integrated values into the correct class structure
202         newEci = new UNIVERSE.ECICoordinates(
203             updatedState[0],
204             updatedState[1],
205             updatedState[2],
206             updatedState[3],
207             updatedState[4],
208             updatedState[5],
209             updatedState[6],
210             updatedState[7],
211             updatedState[8]
212         ); //ECICoordinates
213 
214         return newEci;
215     }
