 47 	/**
 48 		Add the Earth at the center of the Universe
 49 		@public
 50 		@param {Object} earthOptions - .image=the URL of the Earth image to use
 51 	*/
 52     this.addEarth = function(earthOptions) {
 53         var earthSphereSegments = 40, earthSphereRings = 30;
 54 
 55         // Create the sphere
 56         var geometry = new THREE.SphereGeometry(earthSphereRadius, earthSphereSegments, earthSphereRings);
 57 
 58         // Define the material to be used for the sphere surface by pulling the image and wrapping it around the sphere
 59         var shader = {
 60             uniforms : {
 61                 'texture' : {
 62                     type : 't',
 63                     value : 0,
 64                     texture : null
 65                 }
 66             },
 67             vertexShader : ['varying vec3 vNormal;', 'varying vec2 vUv;', 'void main() {', 'gl_Position = projectionMatrix * modelViewMatrix * vec4( position, 1.0 );', 'vNormal = normalize( normalMatrix * normal );', 'vUv = uv;', '}'].join('\n'),
 68             fragmentShader : ['uniform sampler2D texture;', 'varying vec3 vNormal;', 'varying vec2 vUv;', 'void main() {', 'vec3 diffuse = texture2D( texture, vUv ).xyz;', 'float intensity = 1.05 - dot( vNormal, vec3( 0.0, 0.0, 1.0 ) );', 'vec3 atmosphere = vec3( 1.0, 1.0, 1.0 ) * pow( intensity, 3.0 );', 'gl_FragColor = vec4( diffuse + atmosphere, 1.0 );', '}'].join('\n')
 69         };
 70         var uniforms = THREE.UniformsUtils.clone(shader.uniforms);
 71 
 72         uniforms['texture'].texture = THREE.ImageUtils.loadTexture(earthOptions.image);
 73 
 74         var material = new THREE.ShaderMaterial({
 75             uniforms : uniforms,
 76             vertexShader : shader.vertexShader,
 77             fragmentShader : shader.fragmentShader
 78         });
 79 
 80         var earthMesh = new THREE.Mesh(geometry, material);
 81 
 82 		var earthObject = new UNIVERSE.GraphicsObject(
 83 			"earth", 
 84 			"earth", 
 85 			function(elapsedTime) {
 86             	var rotationAngle = CoordinateConversionTools.convertTimeToGMST(universe.getCurrentUniverseTime());
 87             	earthMesh.rotation.y = rotationAngle * (2 * Math.PI / 360);
 88         	},
 89  			function() {
 90                 universe.draw(this.id, earthMesh, false);
 91             });
 92 		universe.addObject(earthObject);
 93     };

219 	/**
220 		Add a Ground Object to the Earth
221 		@public
222 		@param {Object} groundObject - 
223 		id, 
224 	    objectName,
225 	    propagator,
226 	    modelId
227 	*/
228     this.addGroundObject = function(groundObject) {
229         var objectGeometry, objectMaterial, material;
230         if(!groundObject.modelId) {
231             groundObject.modelId = "default_ground_object_geometry";
232             material = "default_ground_object_material";
233         }
234         else {
235             material = "default_material";
236         }
237         universe.getObjectFromLibraryById(groundObject.modelId, function(retrieved_geometry) {
238             objectGeometry = retrieved_geometry;
239             universe.getObjectFromLibraryById(material, function(retrieved_material) {
240                 objectMaterial = retrieved_material;
241                 objectGeometry.applyMatrix( new THREE.Matrix4().setRotationFromEuler( new THREE.Vector3( Math.PI / 2, Math.PI, 0 ) ));
242                 var groundObjectMesh = new THREE.Mesh(objectGeometry, objectMaterial);
243 
244 				var groundGraphicsObject = new UNIVERSE.GraphicsObject(
245 						groundObject.id,
246 						groundObject.objectName,
247 						function(elapsedTime) {
248 	                        // check earth rotation and update location
249 	                        var position = eciTo3DCoordinates(groundObject.propagator());
250 	                        groundObjectMesh.position.set(position.x, position.y, position.z);
251 	                        this.currentLocation = {x: position.x, y: position.y, z: position.z};
252 
253 	                        //http://mrdoob.github.com/three.js/examples/misc_lookat.html
254 	                        var scaled_position_vector = new THREE.Vector3(position.x, position.y, position.z);
255 
256 	                        // arbitrary size, just a point along the position vector further out for the object to lookAt
257 	                        scaled_position_vector.multiplyScalar(1.4);
258 
259 	                        groundObjectMesh.lookAt(scaled_position_vector);
260 	                    },
261 	                    function() {
262 	                        universe.draw(this.id, groundObjectMesh, true);
263 	                    }
264 					);
265 				groundGraphicsObject.currentLocation = undefined;
266 				universe.addObject(groundGraphicsObject);
267             });
268         });
269     };

313 		Add a Propagation Line for an Object
314 		@public
315 		@param {Object} object - 
316 		id, 
317 	    objectName,
318 	    propagator
319 	*/
320     this.addPropogationLineForObject = function(object) {
321         var objectGeometry, objectMaterial;
322         objectGeometry = new THREE.Geometry();
323         universe.getObjectFromLibraryById("default_orbit_line_material", function(retrieved_material) {
324             objectMaterial = retrieved_material;
325             var timeToPropogate = new Date(universe.getCurrentUniverseTime());
326             var loopCount = 1440;
327 
328             // draw a vertex for each minute in a 24 hour period
329             // dropped this to a vertex for every 5 minutes.  This seems to be about the max that you can use for a LEO
330             // and still look decent.  HEOs and GEOs look fine with much greater spans.  For performance reasons, may want
331             // to make this a param that can be set per vehicle
332             for(var j = 0; j < loopCount; j += 5) {
333                 var convertedLocation = eciTo3DCoordinates(object.propagator(timeToPropogate, false));
334                 if(convertedLocation != undefined) {
335                     var vector = new THREE.Vector3(convertedLocation.x, convertedLocation.y, convertedLocation.z);
336                     objectGeometry.vertices.push(new THREE.Vertex(vector));
337                 }
338 
339                 timeToPropogate.setMinutes(timeToPropogate.getMinutes() + 5);
340             }
341 
342             var lineS = new THREE.Line(objectGeometry, objectMaterial);
343 
344 			var lineGraphicsObject = new UNIVERSE.GraphicsObject(
345 				object.id + "_propogation",
346 				object.objectName,
347 				function(elapsedTime) {
348                 // add points onto the end of the track?
349                 },
350 				function() {
351                     universe.draw(this.id, lineS, false);
352                 }
353 			);
354 			universe.addObject(lineGraphicsObject);
355         });
356     }

487 	/**
488 		Return the closest Ground Object to a location
489 		@public
490 		@param {Object} location - x, y, z
491 	*/
492     this.findClosestGroundObject = function(location) {
493 		// TODO: this undefined check may be covering up a bug where not everything gets removed in the 
494 		// removeAllExceptEarthAndMoon method
495 		if(location != undefined) {
496 			var location_vector = new THREE.Vector3(location.x, location.y, location.z);
497 
498 	        // move the vector to the surface of the earth
499 	        location_vector.multiplyScalar(earthSphereRadius / location_vector.length())
500 
501 	        return earthExtensions.findClosestObject({x: location_vector.x, y: location_vector.y, z: location_vector.z});
502 		}
503         return undefined;
504     }

506 	/**
507 		Return the closest Object to a location
508 		@public
509 		@param {Object} location - x, y, z
510 	*/
511     this.findClosestObject = function(location) {
512         var graphicsObjects = universe.getGraphicsObjects();
513         
514         var closestDistance = undefined;
515         var closestObject = undefined;
516 		var location_vector = new THREE.Vector3(location.x, location.y, location.z);
517         
518         for(var i in graphicsObjects) {
519             if(graphicsObjects[i].currentLocation != undefined) {
520                 var vector = new THREE.Vector3(graphicsObjects[i].currentLocation.x, graphicsObjects[i].currentLocation.y, graphicsObjects[i].currentLocation.z);
521                 var distance_to = vector.distanceTo(location_vector);
522                 if(closestDistance == undefined || distance_to < closestDistance) {
523                     closestObject = graphicsObjects[i];
524                     closestDistance = distance_to;
525                 }
526             }
527         }
528         
529         return closestObject;
530     }

667 	/**
668 		Converts ECI to THREE.js 3D coordinate system. Compare these two websites for details on why we have to do this:
669     	http://celestrak.com/columns/v02n01/
670     	http://stackoverflow.com/questions/7935209/three-js-3d-coordinates-system
671 		@private
672 	*/
673     function eciTo3DCoordinates(location) {
674         if(location == undefined) {
675             return undefined;
676         }
677         return {
678             x : -location.x,
679             y : location.z,
680             z : location.y
681         };
682     }
