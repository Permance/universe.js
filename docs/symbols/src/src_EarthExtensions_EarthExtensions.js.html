  5 /** 
  6     Extensions for doing Earth-based 3D modeling with Universe.js

 63     /**
 64         Add a Space Object to the Universe
 65         @public
 66         @param {UNIVERSE.SpaceObject} spaceObject - An orbiting object to add to the Universe
 67         @param {function} callback - A function called at completion of the addition
 68     */
 69     this.addSpaceObject = function (spaceObject, callback) {
 70         var objectGeometry, material;
 71         universe.getObjectFromLibraryById(spaceObject.modelId, function (retrieved_geometry) {
 72             objectGeometry = retrieved_geometry;
 73             universe.getObjectFromLibraryById("default_material", function (retrieved_material) {
 74                 universe.addObject(spaceObject.getGraphicsObject(retrieved_material, objectGeometry, universe, earthExtensions));
 75                 universe.updateOnce();
 76                 callback();
 77             });
 78         });

123     /**
124         Add a non-moving ground point to the universe. This creates a new UNIVERSE.GroundObject for 
125         you, using the LLA coordinates given as parameters.
126         Note that "ground point" can actually be
127         off the surface of the earth (or, under, really). This function quickly lets you set
128         various colors and sizes for the dot.
129         @public
130         @param {object} id - a unique identifier for the dot
131         @param {string} name - a human-readable name for the dot
132         @param {number} color - a hex value RGB color of the dot
133         @param {integer} size - a size for the dot
134         @param {number} lat - the LLA latitude coordinate
135         @param {number} lon - the LLA longitude coordinate
136         @param {number} alt - the LLA altitude coordinate
137         @param {function} callback - a function called once the dot has been added
138      */
139     this.addStaticGroundDot = function (id, name, color, size, lat, lon, alt, callback) {
140         var groundObject = new UNIVERSE.GroundObject(id, name, null, function () {
141             return CoordinateConversionTools.convertLLAtoECI(
142                 new UNIVERSE.LLACoordinates(lat, lon, alt),
143                 CoordinateConversionTools.convertTimeToGMST(universe.getCurrentUniverseTime())
144             );
145         });
146         
147         this.addGroundDot(groundObject, color, size, callback);
148     };

150     /**
151         Add an existing {UNIVERSE.GroundObject} as a static "ground dot".
152         @public
153         @param {UNIVERSE.GroundObject} groundObject - the GroundObject to be placed
154         @param {number} color - the Hex RGB color to display the ground object with
155         @param {integer} size - the size of the "dot"
156         @param {function} callback - a function to be called when the dot is added
157      */
158     this.addGroundDot = function (groundObject, color, size, callback) {
159         var groundObjectGeometryString,
160             groundObjectMaterialString = "dot_" + color;
161 
162         if (size) {
163             groundObjectGeometryString = "ground_dot_size_" + size;
164         } else {
165             groundObjectGeometryString = "default_ground_object_geometry";
166         }
167 
168         try {
169             universe.getObjectFromLibraryById(groundObjectGeometryString, function (retrieved_geometry) {
170                 try {
171                     universe.getObjectFromLibraryById(groundObjectMaterialString, function (retrieved_material) {
172                         universe.addObject(groundObject.getGraphicsObject(retrieved_material, retrieved_geometry, universe, earthExtensions));
173                         universe.updateOnce();
174                         callback();
175                     });
176                 } catch (err) {
177                     // the object wasn't in the library so add it and try to add the dot again'
178                     universe.setObjectInLibrary(groundObjectMaterialString, new THREE.MeshBasicMaterial({
179                         color : color
180                     }));
181                     earthExtensions.addGroundDot(groundObject, color, size, callback);
182                 }
183             });
184         } catch (err) {
185             // the object wasn't in the library so add it and try to add the dot again'
186             universe.setObjectInLibrary(groundObjectGeometryString, new THREE.SphereGeometry(size, size / 10, size / 20));
187             earthExtensions.addGroundDot(groundObject, color, size, callback);
188         }
189     };

253     this.addSensorProjections = function (spaceObject, callback) {
254         var i;

296     */
297     this.addClosestGroundObjectTracingLine = function (object) {
298         var closestObject_id,
299             closestGroundObjectLineController = new UNIVERSE.GraphicsObject(
300                 object.id + "_controlLine",
301                 object.objectName,
302                 undefined,
303                 function (elapsedTime) {
304                     var objectLocation = Utilities.eciTo3DCoordinates(object.propagator(undefined, false), earthExtensions),
305                         closestGroundObject = earthExtensions.findClosestGroundObject(objectLocation);

308                         earthExtensions.removeLineBetweenObjects(object.id, closestObject_id);
309                         closestObject_id = closestGroundObject.id;
310                         earthExtensions.addLineBetweenObjects(object.id, closestObject_id);
311                     }
312                 },
313                 function () {
314 
315                 }
316             );
317         universe.addObject(closestGroundObjectLineController);
318         universe.updateOnce();
319     };

363     /**
364         Return the closest Ground Object to a location
365         @public
366         @param {UNIVERSE.ECICoordinates} location - the location to find the closest point to
367     */
368     this.findClosestGroundObject = function (location) {
369         // TODO: this undefined check may be covering up a bug where not everything gets removed in the 
370         // removeAllExceptEarthAndMoon method
371         if (location) {
372             var location_vector = new THREE.Vector3(location.x, location.y, location.z);
373 
374             // move the vector to the surface of the earth
375             location_vector.multiplyScalar(Constants.radiusEarth / location_vector.length());
376 
377             return earthExtensions.findClosestObject({
378                 x: location_vector.x,
379                 y: location_vector.y,
380                 z: location_vector.z
381             });
382         }
383         return undefined;
384     };
385 
386     /**
387         Return the closest Object to a location
388         @public
389         @param {UNIVERSE.ECICoordinates} location - the location to find the closest point to
390     */
391     this.findClosestObject = function (location) {
392         var graphicsObjects = universe.getGraphicsObjects(),
393             closestDistance,
394             closestObject,
395             location_vector = new THREE.Vector3(location.x, location.y, location.z),
396             i,
397             vector,
398             distance_to;
399 
400         for (i in graphicsObjects) {
401             if (graphicsObjects[i].currentLocation) {
402                 vector = new THREE.Vector3(graphicsObjects[i].currentLocation.x, graphicsObjects[i].currentLocation.y, graphicsObjects[i].currentLocation.z);
403                 distance_to = vector.distanceTo(location_vector);
404                 if (closestDistance === undefined || distance_to < closestDistance) {
405                     closestObject = graphicsObjects[i];
406                     closestDistance = distance_to;
407                 }
408             }
409         }
410 
411         return closestObject;
412     };
