  2 /*global THREE, Utilities */
  3 
  4 var UNIVERSE = UNIVERSE || {};
  5 
  6 /** 
  7     A Ground Object to be drawn on the Earth
  8     @constructor
  9     @param {string} id - Identifier for the object to be referenced later
 10     @param {string} objectName - A name for the object if different than id.  Set to the id if not defined
 11     @param {function} propagator - A function(time) to give the object's position at a time.  No time passed in means the current Universe time
 12     @param {string} modelId - Identifier for the model to use that has been added to the Universe's object library
 13  */
 14 
 15 UNIVERSE.GroundObject = function (id, objectName, modelId, propagator) {
 16     if (!id) {
 17         return undefined;
 18     }
 19     this.id = id;
 20     this.objectName = objectName || id;
 21     this.propagator = propagator;
 22     this.modelId = modelId;
 23 
 24     return this;
 25 };
 26 
 27 UNIVERSE.GroundObject.prototype = {
 28     constructor: UNIVERSE.GroundObject,
 29 
 30     set: function (id, objectName, propagator, modelId) {
 31 
 32         this.id = id;
 33         this.objectName = objectName || id;
 34         this.propagator = propagator;
 35         this.modelId = modelId;
 36 
 37         return this;
 38     },
 39 
 40     getGraphicsObject: function (objectMaterial, objectGeometry, universe, earthExtensions) {
 41         objectGeometry.applyMatrix(new THREE.Matrix4().setRotationFromEuler(new THREE.Vector3(Math.PI / 2, Math.PI, 0)));
 42 
 43         var groundObjectMesh = new THREE.Mesh(objectGeometry, objectMaterial),
 44             groundObject = this,
 45 
 46             groundGraphicsObject = new UNIVERSE.GraphicsObject(
 47                 this.id,
 48                 this.objectName,
 49                 undefined,
 50                 function (elapsedTime) {
 51                     // check earth rotation and update location
 52                     var propagatedPosition = groundObject.propagator(),
 53                         position = Utilities.eciTo3DCoordinates(propagatedPosition, earthExtensions),
 54                         //http://mrdoob.github.com/three.js/examples/misc_lookat.html
 55                         scaled_position_vector = new THREE.Vector3(position.x, position.y, position.z);
 56 
 57                     groundObjectMesh.position.set(position.x, position.y, position.z);
 58                     this.currentLocation = propagatedPosition;
 59 
 60                     // arbitrary size, just a point along the position vector further out for the object to lookAt
 61                     scaled_position_vector.multiplyScalar(1.4);
 62 
 63                     groundObjectMesh.lookAt(scaled_position_vector);
 64                 },
 65                 function () {
 66                     universe.draw(groundObject.id, groundObjectMesh, true);
 67                 }
 68             );
 69 
 70         return groundGraphicsObject;
 71     }
 72 };
