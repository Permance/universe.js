  2 /*global THREE, Utilities */
  3 
  4 // SensorProjection.js
  5 
  6 /**
  7  *
  8  */
  9 var UNIVERSE = UNIVERSE || {};
 10 
 11 UNIVERSE.SensorProjection = function (sensor, object, universe, earthExtensions, objectLocation) {
 12     // Create a SensorPattern
 13 
 14     var sensorPointCount = 30,
 15         // obtain the points of the sensor
 16         points = sensor.buildPointsToDefineSensorShapeInECI(sensorPointCount, object),
 17         extendedPoints = sensor.findProjectionPoints(points, object, 1000),
 18         THREEPoints = new Array(extendedPoints.length),
 19         pointCount = extendedPoints.length,
 20         j,
 21         coord,
 22         objectGeometry,
 23         objectMaterial,
 24         sensorProjection,
 25         sensorProjectionGraphicsObject;
 26 
 27     for (j = 0; j < pointCount; j += 1) {
 28         coord = Utilities.eciTo3DCoordinates(extendedPoints[j], earthExtensions);
 29         THREEPoints[j] = coord;
 30     }
 31 
 32     objectGeometry = new UNIVERSE.SensorProjectionGeometry(objectLocation, THREEPoints);
 33 
 34     objectMaterial = new THREE.MeshBasicMaterial({
 35         color: earthExtensions.defaultObjects.sensorColors.nextColor(),
 36         transparent: true,
 37         blending: THREE.AdditiveBlending,
 38         opacity: 0.15,
 39         overdraw: true
 40     });
 41 
 42     sensorProjection = new THREE.Mesh(objectGeometry, objectMaterial);
 43 
 44     sensorProjection.doubleSided = true;
 45 
 46     sensorProjectionGraphicsObject = new UNIVERSE.GraphicsObject(
 47         object.id + "_sensorProjection_" + sensor.name,
 48         object.objectName,
 49         undefined,
 50         function (elapsedTime) {
 51             if (earthExtensions.enableSensorProjections) {
 52                 var objectLocation = Utilities.eciTo3DCoordinates(object.propagator(undefined, false), earthExtensions),
 53                     points,
 54                     extendedPoints,
 55                     j,
 56                     coord;
 57 
 58                 if (objectLocation) {
 59                     points = sensor.buildPointsToDefineSensorShapeInECI(sensorPointCount, object);
 60                     extendedPoints = sensor.findProjectionPoints(points, object, 1000);
 61 
 62                     THREEPoints = [];
 63                     for (j = 0; j < pointCount; j += 1) {
 64                         coord = Utilities.eciTo3DCoordinates(extendedPoints[j], earthExtensions);
 65                         THREEPoints[j] = coord;
 66                     }
 67                     sensorProjection.geometry.recalculateVertices(objectLocation, THREEPoints);
 68                 }
 69             }
 70         },
 71         function () {
 72             universe.draw(this.id, sensorProjection, false);
 73         }
 74     );
 75 
 76     return sensorProjectionGraphicsObject;
 77 };
