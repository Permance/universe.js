  3 var UNIVERSE = UNIVERSE || {};
  4 
  5 UNIVERSE.UniverseController = function (theRefreshRate) {
  6     var graphicsObjects = [],
  7 
  8         // Timeout that runs the animation, will be cleared when paused
  9         refreshTimeout,
 10 
 11         // number of milliseconds between calls to update() (frame rate / refresh rate)
 12         refreshRate = theRefreshRate || 30,
 13 
 14         // the last time we called update() in ms since jsDate epoch
 15         lastUpdateMs = 0;
 16 
 17     function update() {
 18         // determine how much time has elapsed since update has last been called
 19         var nowMs = (new Date()).getTime(),
 20             elapsedTime = nowMs - lastUpdateMs,
 21             i;
 22         // save now as the last time we've updated
 23         lastUpdateMs = nowMs;
 24 
 25         // causes terrible performance... only enable if needed for debugging!
 26         // logger.debug("now [" + nowMs + "] elapsed ms [" + elapsedTime + "]");
 27         // update and draw all graphics objects
 28         for (i in graphicsObjects) {
 29             graphicsObjects[i].update(elapsedTime);
 30             graphicsObjects[i].draw();
 31         }
 32 
 33         // call update() again in a certain number of milliseconds
 34         refreshTimeout = setTimeout(function () {
 35             update();
 36         }, refreshRate);
 37     }
 38 
 39     this.updateOnce = function () {
 40         var i;
 41         for (i in graphicsObjects) {
 42             graphicsObjects[i].update(null);
 43             graphicsObjects[i].draw();
 44         }
 45     };
 46 
 47     // id
 48     // objectName
 49     // updateFunction
 50     this.addGraphicsObject = function (graphicsObject) {
 51         graphicsObjects[graphicsObject.id] = graphicsObject;
 52         //this.updateOnce();
 53     };
 54 
 55     this.removeGraphicsObject = function (id) {
 56         delete graphicsObjects[id];
 57     };
 58 
 59     this.play = function () {
 60         // set our last update time to now since this is the first update
 61         lastUpdateMs = (new Date()).getTime();
 62         update();
 63     };
 64 
 65     this.pause = function () {
 66         clearTimeout(refreshTimeout);
 67     };
 68 
 69     this.removeAllGraphicsObjects = function () {
 70         graphicsObjects = [];
 71     };
 72 
 73     this.getGraphicsObjects = function () {
 74         return graphicsObjects;
 75     };
 76 
 77     this.getGraphicsObjectById = function (id) {
 78         return graphicsObjects[id];
 79     };
 80 };
