  5 /** 
  6     A Ground Track (Sub-satellite) point to be drawn on the Earth
  7     @constructor
  8     @param {UNIVERSE.SpaceObject} object - the object to draw a ground tracking point for
  9     @param {UNIVERSE.Universe} universe - a Universe instance to draw the ground track point in
 10     @param {UNIVERSE.EarthExtensions} earthExtensions - An EarthExtensions instance to draw the ground track point in
 11     @param {THREE.Material} material - Material for the ground track point
 12     @param {THREE.Geometry} geometry - Geometry for the ground track point
 13  */
 14 var UNIVERSE = UNIVERSE || {};
 15 
 16 UNIVERSE.GroundTrackPoint = function (object, universe, earthExtensions, material, geometry) {
 17 
 18     var groundObjectMesh = new THREE.Mesh(geometry, material),
 19         groundGraphicsObject = new UNIVERSE.GraphicsObject(
 20             object.id + "_groundPoint",
 21             object.objectName,
 22             undefined,
 23             function (elapsedTime) {
 24                 //if(enableSubSatellitePoints) {
 25                 var propagatedLocation = object.propagator(undefined, false),
 26                     objectLocation = Utilities.eciTo3DCoordinates(propagatedLocation, earthExtensions),
 27                     vector;
 28                 if (objectLocation) {
 29                     vector = new THREE.Vector3(objectLocation.x, objectLocation.y, objectLocation.z);
 30 
 31                     // move the vector to the surface of the earth
 32                     vector.multiplyScalar(Constants.radiusEarth / vector.length());
 33 
 34                     groundObjectMesh.position.copy(vector);
 35                 }
 36                 this.currentLocation = propagatedLocation;
 37 
 38             //}
 39             },
 40             function () {
 41                 universe.draw(this.id, groundObjectMesh, true);
 42             }
 43         );
 44 
 45     return groundGraphicsObject;
 46 };
