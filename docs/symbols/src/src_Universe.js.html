  6 /** 
  7     A simple Universe for drawing 3D modeling and simulation using WebGL
  8     @constructor
  9     @param {Date} time - The current universe time
 10     @param {double} refreshRate - The refresh rate for the universe in milliseconds
 11     @param {DOMElement} container - the container where the Universe will be drawn
 12  */
 13 UNIVERSE.Universe = function (time, refreshRate, container) {
 14     var controller = new UNIVERSE.UniverseController(refreshRate),
 15         core = new UNIVERSE.Core3D(container),
 16         objectLibrary = new UNIVERSE.ObjectLibrary(),
 17         currentUniverseTime = time,  // options
 18         playbackSpeed = 1,
 19         stateChangedCallback = function () {}, //function to call when we have a new state object
 20         timeBetweenStateUpdatesMs = 1000, // milliseconds between updating our state object that we broadcast to any listeners
 21         updateStateTimeout, // timeout for updating state
 22         universe = this;
 23 
 24     // OBJECT LIBRARY DEFAULTS
 25 
 26     objectLibrary.setObject("default_geometry", new THREE.Geometry());
 27     objectLibrary.setObject("default_material", new THREE.MeshBasicMaterial());
 28 
 29     // PRIVATE METHODS
 30 
 31     /**
 32         fires a state changed event to the callback
 33         @private
 34     */
 35     function fireStateChanged(state) {
 36         if (stateChangedCallback !== null) {
 37             stateChangedCallback(state);
 38         }
 39     }
 40 
 41     /**
 42         adds an object that updates the currentUniverseTime using the playback speed
 43         @private
 44     */
 45     function addSimStateObject() {
 46         controller.addGraphicsObject({
 47             id : "simState",
 48             objectName : "simState",
 49             update : function (elapsedTime) {
 50                 if (elapsedTime !== null) {
 51                     currentUniverseTime.setTime(currentUniverseTime.getTime() + playbackSpeed * elapsedTime);
 52                 }
 53             },
 54             draw : function () {
 55             }
 56         });
 57     }
 58 
 59     /**
 60         gets called at our state update interval and fires the state change callback
 61         @private
 62     */
 63     function updateState() {
 64         //create our state object and notify our listener
 65         var state = {
 66             currentUniverseTime: new Date(currentUniverseTime)
 67         };
 68 
 69         fireStateChanged(state);
 70 
 71         // call update() again in a certain number of milliseconds
 72         updateStateTimeout = setTimeout(function () {
 73             updateState();
 74         }, timeBetweenStateUpdatesMs);
 75     }
 76 
 77     // PROTECTED METHODS (API METHODS)
 78 
 79     /**
 80         Start playback for the universe
 81         @public
 82         @param {date} startTime
 83         @param {double} newPlaybackSpeed
 84         @param {function} newStateChangedCallback
 85      */
 86     this.play = function (startTime, newPlaybackSpeed, newStateChangedCallback) {
 87         if (startTime) {
 88             currentUniverseTime = new Date(startTime);
 89         }
 90         if (newPlaybackSpeed) {
 91             playbackSpeed = newPlaybackSpeed;
 92         }
 93 
 94         if (newStateChangedCallback) {
 95             stateChangedCallback = newStateChangedCallback;
 96         }
 97 
 98         // update state our first time
 99         updateState();
100 
101         controller.play();
102     };

163     this.addJsonGeometryModel = function (modelId, modelUrl, callback) {

231     /**
232         Snap the Universe's camera to be directly behind an object
233         @public
234         @param {string} id - identifier for the object
235     */
236     this.snapToObject = function (id) {
237         // get the object's position and copy it into a vector
238         var position = core.getObjectPosition(id),
239             vector;
240         if (position) {
241             vector = new THREE.Vector3();
242             vector.copy(position);
243 
244             // move the point the camera will be at out a bit so we are behind the object
245             vector.multiplyScalar(1.4);
246 
247             // tell the core to move to the vector
248             core.moveCameraTo(vector);
249         } // else Object is not added to the core so don't do anything
250     };
