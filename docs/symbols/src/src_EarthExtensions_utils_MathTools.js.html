  5      * returns the angle between two vectors
  6      * @param {vector} x (unitless)
  7      * @param {vector} y (unitless)
  8      *
  9      * @returns {Number} angle
 10      */
 11     angleBetweenTwoVectorsVector: function (x, y) {
 12         var angle,                 //double
 13             magX = x.length(),       //double
 14             magY = y.length(),       //double
 15             xDotY = x.dot(y); //double
 16 
 17         angle = MathTools.toDegrees(Math.acos(xDotY / (magX * magY)));
 18 
 19         if (angle > 90) {
 20             angle = 180 - angle;
 21         }
 22 
 23         return angle; //deg
 24     },

 26     /**
 27      * rotates vector 'vec' about the x-axis by 'x' degrees
 28      * @param {Number} x (degrees)
 29      * @param {Double[]} vec (an array of length 3, assumed to be x,y,z values)
 30      *
 31      * @returns {Double[]} result (3x1)
 32      */
 33     rot1: function (x, vec) {
 34         x = MathTools.toRadians(x);
 35 
 36         var result = []; //Double[3];
 37         result[0] = vec[0];
 38         result[1] = Math.cos(x) * vec[1] + Math.sin(x) * vec[2];
 39         result[2] = -Math.sin(x) * vec[1] + Math.cos(x) * vec[2];
 40 
 41         return result;
