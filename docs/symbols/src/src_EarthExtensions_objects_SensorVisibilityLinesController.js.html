  4 // SensorVisibilityLinesController.js
  5 
  6 var UNIVERSE = UNIVERSE || {};
  7 
  8 UNIVERSE.SensorVisibilityLinesController = function (object, universe, earthExtensions) {
  9     var visibilityLinesController = new UNIVERSE.GraphicsObject(
 10         object.id + "_visibilityLines",
 11         object.objectName,
 12         undefined,
 13         function (elapsedTime) {
 14             if (earthExtensions.enableVisibilityLines) {
 15                 var sensorLength = object.sensors.length,
 16                     graphicsObjects = universe.getGraphicsObjects(),
 17                     objectsToDrawLinesTo = [],
 18                     i,
 19                     j,
 20                     k,
 21                     sensor,
 22                     obj,
 23                     inView;
 24                 for (i = 0; i < sensorLength; i += 1) {
 25                     sensor = object.sensors[i];
 26                     for (j in graphicsObjects) {
 27                         obj = graphicsObjects[j];
 28                         if (obj.currentLocation !== undefined &&
 29                                 obj.modelName !== "earth" &&
 30                                 obj.modelName !== "moon" &&
 31                                 obj.modelName !== "sun" &&
 32                                 obj.id !== object.id &&
 33                                 obj.id.indexOf("_groundPoint") === -1 &&
 34                                 obj.id.indexOf("_propagation") === -1 &&
 35                                 obj.id.indexOf("_to_") === -1 &&
 36                                 obj.id.indexOf("_visibility_") === -1) {
 37                             // Now we're looking at a point 
 38 
 39                             inView = sensor.checkSensorVisibilityOfTargetPoint(object, obj.currentLocation);
 40 
 41                             if (!objectsToDrawLinesTo[obj.id]) {
 42                                 objectsToDrawLinesTo[obj.id] = inView;
 43                             }
 44                         }
 45                     }
 46                 }
 47 
 48                 for (k in objectsToDrawLinesTo) {
 49                     if (objectsToDrawLinesTo[k]) {
 50                         if (universe.getGraphicsObjectById(object.id + "_visibility_" + k) === undefined) {
 51                             //console.log("adding line for object: " + object.id + " and " + k);
 52                             earthExtensions.addLineBetweenObjects(object.id, k, undefined, "_visibility_");
 53                         //universe.updateOnce();
 54                         }
 55                     } else {
 56                         earthExtensions.removeLineBetweenObjects(object.id, k, "_visibility_");
 57                     }
 58                 //console.log("finished: " + k);
 59                 }
 60             }
 61             earthExtensions.showAllSensorVisibilityLines(earthExtensions.enableVisibilityLines);
 62         },
 63         function () {
 64         // nothing to draw, this is a controller
 65         }
 66     );
 67 
 68     return visibilityLinesController;
 69 };
