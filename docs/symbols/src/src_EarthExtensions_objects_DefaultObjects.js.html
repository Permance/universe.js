1 /*jslint browser: true, sloppy: true */
  2 /*global UNIVERSE, THREE */
  3 /** 
  4     A set of default objects to add to the Universe Object Library
  5     @constructor
  6     @param {UNIVERSE.Universe} universe - A Universe instance to use the default objects in
  7  */
  8 UNIVERSE.DefaultObjects = function (universe) {
  9     universe.setObjectInLibrary("default_ground_object_geometry", new THREE.SphereGeometry(200, 20, 10));
 10     universe.setObjectInLibrary("default_ground_object_material", new THREE.MeshBasicMaterial({
 11         color : 0xCC0000
 12     }));
 13 
 14     universe.setObjectInLibrary("default_ground_track_material", new THREE.MeshBasicMaterial({
 15         color: 0xCC0000,
 16         transparent: true,
 17         opacity: 0.4,
 18         blending: THREE.AdditiveBlending
 19     }));
 20 
 21     universe.setObjectInLibrary("default_orbit_line_material", new THREE.LineBasicMaterial({
 22         color : 0x990000,
 23         opacity : 1
 24     }));
 25 
 26     universe.setObjectInLibrary("default_ground_object_tracing_line_material", new THREE.LineBasicMaterial({
 27         color : 0x009900,
 28         opacity : 1
 29     }));
 30 
 31     this.sensorColors = {
 32         colorList: [
 33             "0xff0000",
 34             "0x00cc00",
 35             "0x0066ff",
 36             "0x9900cc",
 37             "0xffff00",
 38             "0xff6666",
 39             "0xebebeb",
 40             "0xffaa00"
 41         ],
 42         iterator: -1,
 43         // Grab the next color on the list and iterate to the next color
 44         nextColor: function () {
 45             this.iterator = (this.iterator + 1) % this.colorList.length;
 46             return this.colorList[this.iterator];
 47         }
 48     };
 49 };
